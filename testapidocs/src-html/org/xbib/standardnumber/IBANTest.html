007    public class IBANTest extends Assert {

009        @Test
010        public void testIBAN1() throws Exception {
011            String value = "GB82 WEST 1234 5698 7654 32";
012            IBAN iban = new IBAN().set(value).normalize().verify();
013            assertEquals(iban.normalizedValue(), "GB82WEST12345698765432");
014            assertEquals(iban.format(), "GB82WEST12345698765432");
015        }
016    
017        @Test
018        public void testIBAN2() throws Exception {
019            String value = "NL91ABNA0417164300";
020            IBAN iban = new IBAN().set(value).normalize().verify();
021            assertEquals("NL91ABNA0417164300", iban.normalizedValue());
022            assertEquals("NL91ABNA0417164300", iban.format());
023        }

028                    "SE1212312345678901234561",

030            };
031            for (String value : str) {
032                new IBAN().set(value).normalize().verify();
033            }
