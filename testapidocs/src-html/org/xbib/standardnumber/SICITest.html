002package org.xbib.standardnumber;
003
004import org.testng.Assert;
005import org.testng.annotations.Test;
006
007public class SICITest extends Assert {
008
009    @Test
010    public void testSICI1() throws Exception {
011        SICI sici = new SICI().set("0095-4403(199502/03)21:3<12:WATIIB>2.0.TX;2-J").normalize().verify();
012        assertEquals("0095-4403(199502/03)21:3<12:WATIIB>2.0.TX;2-J", sici.normalizedValue());
013        assertEquals("SICI 0095-4403(199502/03)21:3<12:WATIIB>2.0.TX;2-J", sici.format());
014    }
