001package org.xbib.standardnumber;
002
003import org.testng.Assert;
004import org.testng.annotations.Test;
005
006public class ISSNTest extends Assert {
007
008    @Test
009    public void testISSN() throws Exception {
010        ISSN issn = new ISSN().set("1869-7127").normalize().verify();
011        assertEquals(issn.normalizedValue(), "18697127");
012        assertEquals(issn.format(), "1869-7127");
013        assertEquals(issn.createChecksum(true).toGTIN("03").normalizedValue(), "9771869712038");
014    }
015
016
017}
