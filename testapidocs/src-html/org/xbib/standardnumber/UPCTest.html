002    package org.xbib.standardnumber;
003    
004    import org.testng.Assert;
005    import org.testng.annotations.Test;
006    
007    public class UPCTest extends Assert {
008    
009        @Test
010        public void testUPC() throws Exception {
011            String value = "796030114977";
012            UPC upc = new UPC().set(value).normalize().verify();
013            assertEquals("796030114977", upc.normalizedValue());
014            assertEquals("796030114977", upc.format());
015        }
016    
017        @Test
018        public void testUPC2() throws Exception {
019            String value = "036000291452";
020            UPC upc = new UPC().set(value).normalize().verify();
021            assertEquals("036000291452", upc.normalizedValue());
022            assertEquals("036000291452", upc.format());
023        }
