001    package org.xbib.standardnumber;

003    import org.testng.Assert;
004    import org.testng.annotations.Test;

006    public class ISANTest extends Assert {
007    
008        @Test
009        public void testISAN0() throws Exception {
010            String value = "ISAN B159-D8FA-0124-0000-K";
011            ISAN isan = new ISAN().set(value).createChecksum(true).normalize().verify();
012            assertEquals(isan.normalizedValue(), "B159D8FA01240000K");
013            assertEquals(isan.format(), "ISAN B159-D8FA-0124-0000-K");
014        }
015
