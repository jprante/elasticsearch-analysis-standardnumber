001package org.xbib.standardnumber;
002
003import org.testng.Assert;
004import org.testng.annotations.Test;
005
006public class ISANTest extends Assert {
007
008    @Test
009    public void testISAN0() throws Exception {
010        String value = "ISAN B159-D8FA-0124-0000-K";
011        ISAN isan = new ISAN().set(value).createChecksum(true).normalize().verify();
012        assertEquals(isan.normalizedValue(), "B159D8FA01240000K");
013        assertEquals(isan.format(), "ISAN B159-D8FA-0124-0000-K");
014    }
015
016    @Test
017    public void testISAN1() throws Exception {
018        String value = "ISAN 0000-3BAB-9352-0000-G ";
019        ISAN isan = new ISAN().set(value).createChecksum(true).normalize().verify();
020        assertEquals(isan.normalizedValue(), "00003BAB93520000G");
021        assertEquals(isan.format(), "ISAN 0000-3BAB-9352-0000-G");
022    }
023
024    @Test
025    public void testISAN2() throws Exception {
026        String value = "006A-15FA-002B-C95F-W";
027        ISAN isan = new ISAN().set(value).createChecksum(true).normalize().verify();
028        assertEquals(isan.normalizedValue(), "006A15FA002BC95FW");
029        assertEquals(isan.format(), "ISAN 006A-15FA-002B-C95F-W");
030    }
031
032    @Test
033    public void testISAN3() throws Exception {

037        assertEquals( isan.format(), "ISAN 1881-66C7-3420-0000-3");
038    }
039
040    @Test

045        assertEquals(isan.format(), "ISAN 0001-F54C-302A-8D98-N-0000-0121-O");
046    }

056    @Test

079
080    @Test
081    public void testBrokenISAN() {
082        try {
083            String value = "1435-1838 = Lehrergilde-Rundbrief";
084            new ISAN().set(value).createChecksum(true).normalize().verify();
085            assertTrue(false);
086        } catch (NumberFormatException e) {
087            assertTrue(true);
088        }
089    }
