002package org.xbib.standardnumber;
003
004import org.testng.Assert;
005import org.testng.annotations.Test;
006
007public class ORCIDTest extends Assert {
008
009    @Test
010    public void testORCID() throws Exception {
011        String value = "0000-0002-1825-0097";
012        ORCID orcid = new ORCID().set(value).normalize().verify();
013        assertEquals("0000000218250097", orcid.normalizedValue());
014        assertEquals("0000000218250097", orcid.format());
015        assertEquals("http://orcid.org/0000000218250097", orcid.toURI().toString());
016    }
017
018    @Test
019    public void testORCID2() throws Exception {
020        String value = "0000-0001-5109-3700";

023        assertEquals("0000000151093700", orcid.format());
024        assertEquals("http://orcid.org/0000000151093700", orcid.toURI().toString());
025    }
026
027    @Test
