002    package org.xbib.standardnumber;
003    
004    import org.testng.Assert;
005    import org.testng.annotations.Test;
006    
007    public class ISNITest extends Assert {
008    
009        @Test
010        public void testISNI() throws Exception {
011            String value = "0000-0002-1825-0097";
012            ISNI isni = new ISNI().set(value).normalize().verify();
013            assertEquals("0000000218250097", isni.normalizedValue());
014            assertEquals("0000000218250097", isni.format());
015        }
016    
017        @Test
018        public void testISNI2() throws Exception {
019            String value = "ISNI 1422 4586 3573 0476";
020            ISNI isni = new ISNI().set(value).normalize().verify();
021            assertEquals("1422458635730476", isni.normalizedValue());
022            assertEquals("1422458635730476", isni.format());
023        }
024    
025        @Test
026        public void testISNI3() throws Exception {
027            // fix wrong checksum in MARBI example http://www.loc.gov/marc/marbi/2010/2010-dp03.html
028            String value = "ISNI 8462 8328 5653 6435";
029            ISNI isni = new ISNI().set(value).checksum().normalize().verify();
030            assertEquals("8462832856536436", isni.normalizedValue());
031            assertEquals("8462832856536436", isni.format());
032        }
033    
034        @Test
035        public void testISNI4() throws Exception {
036            // fix wrong checksum in MARBI example http://www.loc.gov/marc/marbi/2010/2010-dp03.html
037            String value = "ISNI 0023 0000 1001 1234";
038            ISNI isni = new ISNI().set(value).checksum().normalize().verify();
039            assertEquals("0023000010011235", isni.normalizedValue());
040            assertEquals("0023000010011235", isni.format());
041        }
042    
043        @Test
044        public void testISNI5() throws Exception {
045            String value = "0000 0001 2195 3271";
046            ISNI isni = new ISNI().set(value).normalize().verify();
047            assertEquals("0000000121953271", isni.normalizedValue());
048            assertEquals("0000000121953271", isni.format());
049        }
050    }
