001package org.xbib.standardnumber;
002
003import org.testng.Assert;
004import org.testng.annotations.Test;
005
006public class DOITest extends Assert {
007
008    @Test
009    public void testDOI() throws Exception {
010        DOI doi = new DOI().set("10.1016/0032-3861(93)90481-o").normalize().verify();
011        assertEquals(doi.normalizedValue(), "10.1016/0032-3861(93)90481-o");
012        assertEquals(doi.format(), "http://doi.org/10.1016/0032-3861(93)90481-o");
013    }
014
015
016}
