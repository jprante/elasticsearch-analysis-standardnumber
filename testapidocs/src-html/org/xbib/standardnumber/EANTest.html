002    package org.xbib.standardnumber;
003    
004    import org.testng.Assert;
005    import org.testng.annotations.Test;
006    
007    public class EANTest extends Assert {
008    
009        @Test
010        public void testEAN() throws Exception {
011            String value = "4 007630 000110";
012            EAN ean = new EAN().set(value).checksum().normalize().verify();
013            assertEquals("4007630000116", ean.normalizedValue());
014            assertEquals("4007630000116", ean.format());
015        }
016    
017        @Test
018        public void testEAN2() throws Exception {
019            String value = "7501031311309";
020            EAN ean = new EAN().set(value).checksum().normalize().verify();
021            assertEquals("7501031311309", ean.normalizedValue());
022            assertEquals("7501031311309", ean.format());
023        }
024    
025    }
