002    package org.xbib.standardnumber;
003    
004    import org.testng.Assert;
005    import org.testng.annotations.Test;
006    
007    public class ISBNTest extends Assert {
008    
009        @Test
010        public void testDehypenate() {
011            assertEquals(new ISBN().set("000-111-333").normalize().normalizedValue(), "000111333");
012        }
013    
014        @Test(expectedExceptions = IllegalArgumentException.class)
015        public void testISBNTooShort() throws Exception {
016            new ISBN().set("12-7").normalize().verify();
017        }
018    
019        @Test
020        public void testDirtyISBN() throws Exception {
021            String value = "ISBN 3-9803350-5-4 kart. : DM 24.00";
022            ISBN isbn = new ISBN().set(value).normalize().verify();
023            assertEquals(isbn.normalizedValue(), "3980335054");
024        }
025    
026        @Test(expectedExceptions = NumberFormatException.class)
027        public void testTruncatedISBN() throws Exception {
028            String value = "ISBN";
029            new ISBN().set(value).normalize().verify();
030        }
031    
032        @Test
033        public void fixChecksum() throws Exception {
034            String value = "3616065810";
035            ISBN isbn = new ISBN().set(value).checksum().normalize().verify();
036            assertEquals(isbn.normalizedValue(), "361606581X");
037        }
038    
039        @Test
040        public void testEAN() throws Exception {
041            String value = "978-3-551-75213-0";
042            ISBN ean = new ISBN().set(value).ean(true).normalize().verify();
043            assertEquals(ean.normalizedValue(), "9783551752130");
044            assertEquals(ean.format(), "978-3-551-75213-0");
045        }
046    
047        @Test
048        public void testEAN2() throws Exception {
049            String value = "978-3-551-75213-1";
050            ISBN ean = new ISBN().set(value).ean(true).checksum().normalize().verify();
051            assertEquals(ean.normalizedValue(), "9783551752130");
052            assertEquals(ean.format(), "978-3-551-75213-0");
053        }
054    
055        @Test(expectedExceptions = NumberFormatException.class)
056        public void testWrongAndDirtyEAN() throws Exception {
057            // correct ISBN-10 is 3-451-04112-X
058            String value = "ISBN ISBN 3-451-4112-X kart. : DM 24.80";
059            new ISBN().set(value).ean(false).checksum().normalize().verify();
060        }
061    }
