001    
002    package org.xbib.standardnumber;
003    
004    import org.testng.Assert;
005    import org.testng.annotations.Test;
006    
007    public class LCCNTest extends Assert {
008    
009        /*
010         *  "n78-890351" normalizes to "n78890351".
011         *  "n78-89035" normalizes to "n78089035".
012         *  "n 78890351 " normalizes to "n78890351".
013         *  " 85000002 " normalizes to "85000002"
014         *  "85-2 " normalizes to "85000002"
015         *  "2001-000002" normalizes to "2001000002"
016         *  "75-425165//r75" normalizes to "75425165"
017         *  " 79139101 /AC/r932" normalizes to "79139101"
018         */
019    
020        @Test
021        public void normalizeLCCN1() throws Exception {
022            String s = "n78-890351";
023            LCCN lccn = new LCCN().set(s).normalize().verify();
024            assertEquals(lccn.format(), "n78890351");
025        }
026    
027        @Test
028        public void normalizeLCCN2() throws Exception {
029            String s = "n78-89035";
030            LCCN lccn = new LCCN().set(s).normalize().verify();
031            assertEquals(lccn.format(), "n78089035");
032        }
033    
034        @Test
035        public void normalizeLCCN3() throws Exception {
036            String s = "n 78890351 ";
037            LCCN lccn = new LCCN().set(s).normalize().verify();
038            assertEquals(lccn.format(), "n78890351");
039        }
040    
041        @Test
042        public void normalizeLCCN4() throws Exception {
043            String s = " 85000002 ";
044            LCCN lccn = new LCCN().set(s).normalize().verify();
045            assertEquals(lccn.format(), "85000002");
046        }
047    
048        @Test
049        public void normalizeLCCN5() throws Exception {
050            String s = "85-2";
051            LCCN lccn = new LCCN().set(s).normalize().verify();
052            assertEquals(lccn.format(), "85000002");
053        }
054    
055        @Test
056        public void normalizeLCCN6() throws Exception {
057            String s = "2001-000002";
058            LCCN lccn = new LCCN().set(s).normalize().verify();
059            assertEquals(lccn.format(), "2001000002");
060        }
061    
062        @Test
063        public void normalizeLCCN7() throws Exception {
064            String s = "75-425165//r75";
065            LCCN lccn = new LCCN().set(s).normalize().verify();
066            assertEquals(lccn.format(), "75425165");
067        }
068    
069        @Test
070        public void normalizeLCCN8() throws Exception {
071            String s = " 79139101 /AC/r932";
072            LCCN lccn = new LCCN().set(s).normalize().verify();
073            assertEquals(lccn.format(), "79139101");
074        }
075    }
