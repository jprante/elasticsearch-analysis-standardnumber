001    package org.xbib.standardnumber;
002    
003    import org.testng.Assert;
004    import org.testng.annotations.Test;
005    
006    public class ISTCTest extends Assert {
007    
008        @Test
009        public void testISTC() throws Exception {
010            ISTC istc = new ISTC().set("0A920021223F3320").normalize().verify();
011            assertEquals("0A920021223F3320", istc.normalizedValue());
012            assertEquals("ISTC 0A9-2002-1223F332-0", istc.format());
013        }
014    
015        @Test
016        public void testISTC2() throws Exception {
017            ISTC istc = new ISTC().set("ISTC A02-2009-000004BE-A").normalize().verify();
018            assertEquals("A022009000004BEA", istc.normalizedValue());
019            assertEquals("ISTC A02-2009-000004BE-A", istc.format());
020        }
021    
022        @Test
023        public void testISTC3() throws Exception {
024            ISTC istc = new ISTC().set("ISTC 0A9 2009 12B4A105 C").normalize().verify();
025            assertEquals("0A9200912B4A105C", istc.normalizedValue());
026            assertEquals("ISTC 0A9-2009-12B4A105-C", istc.format());
027        }
028    
029        @Test
030        public void testISTC4() throws Exception {
031            ISTC istc = new ISTC().set("ISTC 0A9-2002-12B4A105-7").normalize().verify();
032            assertEquals("0A9200212B4A1057", istc.normalizedValue());
033            assertEquals("ISTC 0A9-2002-12B4A105-7", istc.format());
034        }
035    
036        @Test
037        public void testISTC5() throws Exception {
038            ISTC istc = new ISTC().set("A02200900000A87C").normalize().verify();
039            assertEquals("A02200900000A87C", istc.normalizedValue());
040            assertEquals("ISTC A02-2009-00000A87-C", istc.format());
041        }
042    
043        @Test
044        public void testISTC6() throws Exception {
045            ISTC istc = new ISTC().set("ISTC A02-2010-31F4CB2C-B").normalize().verify();
046            assertEquals("A02201031F4CB2CB", istc.normalizedValue());
047            assertEquals("ISTC A02-2010-31F4CB2C-B", istc.format());
048        }
049    
050    }
