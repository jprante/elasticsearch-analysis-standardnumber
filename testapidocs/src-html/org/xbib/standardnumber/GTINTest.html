002    package org.xbib.standardnumber;
003    
004    import org.testng.Assert;
005    import org.testng.annotations.Test;
006    
007    public class GTINTest extends Assert {
008    
009        @Test
010        public void testGTIN() throws Exception {
011            String value = "4104420033801";
012            GTIN gtin = new GTIN().set(value).normalize().verify();
013            assertEquals("4104420033801", gtin.normalizedValue());
014            assertEquals("4104420033801", gtin.format());
015        }
016    
017        @Test
018        public void testISSNGTIN() throws Exception {
019            String value = "977-1869712-03-8";
020            GTIN gtin = new GTIN().set(value).normalize().verify();
021            assertEquals("9771869712038", gtin.normalizedValue());
022            assertEquals("9771869712038", gtin.format());
023        }
024    
025        @Test
026        public void testGTIN2() throws Exception {
027            String value = "4191054501707";
028            GTIN gtin = new GTIN().set(value).normalize().verify();
029            assertEquals("4191054501707", gtin.normalizedValue());
030            assertEquals("4191054501707", gtin.format());
031        }
032    
033        @Test
034        public void testISBNGTIN() throws Exception {
035            String value = "9783652002264";
036            GTIN gtin = new GTIN().set(value).normalize().verify();
037            assertEquals("9783652002264", gtin.normalizedValue());

039        }
