001    package org.xbib.elasticsearch.index.analysis;
002    
003    import org.apache.lucene.analysis.TokenStream;
004    import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
005    import org.elasticsearch.common.inject.Injector;
006    import org.elasticsearch.common.inject.ModulesBuilder;
007    import org.elasticsearch.common.logging.ESLogger;
008    import org.elasticsearch.common.logging.ESLoggerFactory;
009    import org.elasticsearch.common.settings.ImmutableSettings;
010    import org.elasticsearch.common.settings.Settings;
011    import org.elasticsearch.common.settings.SettingsModule;
012    import org.elasticsearch.env.Environment;
013    import org.elasticsearch.env.EnvironmentModule;
014    import org.elasticsearch.index.Index;
015    import org.elasticsearch.index.IndexNameModule;
016    import org.elasticsearch.index.analysis.AnalysisModule;
017    import org.elasticsearch.index.analysis.AnalysisService;
018    import org.elasticsearch.index.analysis.NamedAnalyzer;
019    import org.elasticsearch.index.settings.IndexSettingsModule;
020    import org.elasticsearch.indices.analysis.IndicesAnalysisModule;
021    import org.elasticsearch.indices.analysis.IndicesAnalysisService;
022    import org.testng.Assert;
023    import org.testng.annotations.BeforeClass;
024    import org.testng.annotations.Test;
025    import org.xbib.elasticsearch.plugin.analysis.standardnumber.AnalysisStandardNumberPlugin;
026    
027    import java.io.IOException;
028    
029    
030    public class StandardNumberTokenFilterTests extends Assert {
031    
032        private final static ESLogger logger = ESLoggerFactory.getLogger("test");
033    
034        NamedAnalyzer analyzer;
035    
036        @BeforeClass
037        public void create() {
038            AnalysisService analysisService = createAnalysisService();
039            analyzer = analysisService.analyzer("standardnumber");
040        }
041    
042        @Test
043        public void testISBN1() throws IOException {
044             String source = "Die ISBN von Elasticsearch in Action lautet 9781617291623";
045    
046            String[] expected = new String[] {
047                    "Die",
048                    "ISBN",
049                    "von",
050                    "Elasticsearch",
051                    "in",
052                    "Action",
053                    "lautet",
054                    "9781617291623",
055                    "978-1-61729-162-3"
056            };
057            assertSimpleTSOutput(analyzer.tokenStream("content", source), expected);
058        }
059    
060        @Test
061        public void testISBN2() throws IOException {
062            String source = "Die ISBN von Lucene in Action lautet 1-9339-8817-7.";
063    
064            String[] expected = new String[] {
065                    "Die",
066                    "ISBN",
067                    "von",
068                    "Lucene",
069                    "in",
070                    "Action",
071                    "lautet",
072                    "1-9339-8817-7.",
073                    "1-933988-17-7",
074                    "1933988177",
075                    "978-1-933988-17-7",
076                    "9781933988177"
077            };
078            assertSimpleTSOutput(analyzer.tokenStream("content", source), expected);
079        }
080    
081        private AnalysisService createAnalysisService() {
082            Settings settings = ImmutableSettings.settingsBuilder()
083                    .loadFromClasspath("org/xbib/elasticsearch/index/analysis/standardnumber.json").build();
084    
085            Index index = new Index("test");
086    
087            Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
088                    new EnvironmentModule(new Environment(settings)),
089                    new IndicesAnalysisModule())
090                    .createInjector();
091    
092            AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
093            new AnalysisStandardNumberPlugin().onModule(analysisModule);
094    
095            Injector injector = new ModulesBuilder().add(
096                    new IndexSettingsModule(index, settings),
097                    new IndexNameModule(index),
098                    analysisModule)
099                    .createChildInjector(parentInjector);
100    
101            return injector.getInstance(AnalysisService.class);
102        }
103    
104        private static void assertSimpleTSOutput(TokenStream stream,
105                                                 String[] expected) throws IOException {
106            stream.reset();
107            CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
108            assertNotNull(termAttr);
109            int i = 0;
110            while (stream.incrementToken()) {
111                assertTrue(i < expected.length);
112                assertEquals(termAttr.toString(), expected[i++], "expected different term at index " + i);
113            }
114            assertEquals(i, expected.length, "not all tokens produced");
115        }
116    }
