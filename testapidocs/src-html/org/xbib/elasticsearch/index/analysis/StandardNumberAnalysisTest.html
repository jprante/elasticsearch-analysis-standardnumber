001package org.xbib.elasticsearch.index.analysis;
002
003import org.apache.lucene.analysis.TokenStream;
004import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
005import org.elasticsearch.common.inject.Injector;
006import org.elasticsearch.common.inject.ModulesBuilder;
007import org.elasticsearch.common.settings.ImmutableSettings;
008import org.elasticsearch.common.settings.Settings;
009import org.elasticsearch.common.settings.SettingsModule;
010import org.elasticsearch.env.Environment;
011import org.elasticsearch.env.EnvironmentModule;
012import org.elasticsearch.index.Index;
013import org.elasticsearch.index.IndexNameModule;
014import org.elasticsearch.index.analysis.NamedAnalyzer;
015import org.elasticsearch.index.analysis.AnalysisModule;
016import org.elasticsearch.index.analysis.AnalysisService;
017import org.elasticsearch.index.settings.IndexSettingsModule;
018import org.elasticsearch.indices.analysis.IndicesAnalysisModule;
019import org.elasticsearch.indices.analysis.IndicesAnalysisService;
020import org.xbib.elasticsearch.index.analysis.standardnumber.StandardNumberAnalyzer;
021import org.xbib.elasticsearch.plugin.analysis.standardnumber.AnalysisStandardNumberPlugin;
022import org.testng.annotations.Test;
023
024import java.io.IOException;
025
026import static org.testng.Assert.assertEquals;
027import static org.testng.Assert.assertNotNull;
028import static org.testng.Assert.assertTrue;
029
030
031public class StandardNumberAnalysisTest {
032
033    @Test
034    public void testStandardNumberAnalyzer() throws IOException {
035        NamedAnalyzer namedAnalyzer = createAnalysisService().analyzer("standardnumber");
036        String[] expected = new String[]{
037                "Die",
038                "ISBN",
039                "von",
040                "Elasticsearch",
041                "in",
042                "Action",
043                "lautet",
044                "9781617291623",
045                "EAN 9781617291623",
046                "GTIN 9781617291623",
047                "ISBN 978-1-61729-162-3",
048                "ISBN 9781617291623"
049        };
050        assertSimpleTSOutput(namedAnalyzer.tokenStream("content", "Die ISBN von Elasticsearch in Action lautet 9781617291623"), expected);
051    }
052
053    private AnalysisService createAnalysisService() {
054        Settings settings = ImmutableSettings.settingsBuilder().build();
055
056        Index index = new Index("test");
057
058        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
059                new EnvironmentModule(new Environment(settings)),
060                new IndicesAnalysisModule())
061                .createInjector();
062
063        AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
064        new AnalysisStandardNumberPlugin().onModule(analysisModule);
065
066        Injector injector = new ModulesBuilder().add(
067                new IndexSettingsModule(index, settings),
068                new IndexNameModule(index),
069                analysisModule)
070                .createChildInjector(parentInjector);
071
072        return injector.getInstance(AnalysisService.class);
073    }
074
075    private void assertSimpleTSOutput(TokenStream stream, String[] expected) throws IOException {
076        stream.reset();
077        CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
078        assertNotNull(termAttr);
079        int i = 0;
080        while (stream.incrementToken()) {
081            assertTrue(i < expected.length);
082            assertEquals(termAttr.toString(), expected[i++], "expected different term at index " + i);
083        }
084        assertEquals(i, expected.length, "not all tokens produced");
085        stream.close();
086    }
087}
