001package org.xbib.elasticsearch.index.analysis;
002
003import org.apache.lucene.analysis.TokenStream;
004import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
005import org.elasticsearch.common.inject.Injector;
006import org.elasticsearch.common.inject.ModulesBuilder;
007import org.elasticsearch.common.settings.ImmutableSettings;
008import org.elasticsearch.common.settings.Settings;
009import org.elasticsearch.common.settings.SettingsModule;
010import org.elasticsearch.env.Environment;
011import org.elasticsearch.env.EnvironmentModule;
012import org.elasticsearch.index.Index;
013import org.elasticsearch.index.IndexNameModule;
014import org.elasticsearch.index.analysis.AnalysisModule;
015import org.elasticsearch.index.analysis.AnalysisService;
016import org.elasticsearch.index.analysis.NamedAnalyzer;
017import org.elasticsearch.index.settings.IndexSettingsModule;
018import org.elasticsearch.indices.analysis.IndicesAnalysisModule;
019import org.elasticsearch.indices.analysis.IndicesAnalysisService;
020import org.testng.Assert;
021import org.testng.annotations.BeforeClass;
022import org.testng.annotations.Test;
023import org.xbib.elasticsearch.plugin.analysis.standardnumber.AnalysisStandardNumberPlugin;
024
025import java.io.IOException;
026
027
028public class StandardNumberTokenFilterTest extends Assert {
029
030    NamedAnalyzer analyzer;
031
032    @BeforeClass
033    public void create() {
034        AnalysisService analysisService = createAnalysisService();
035        analyzer = analysisService.analyzer("standardnumber");
036    }
037
038    @Test
039    public void testISBN1() throws IOException {
040         String source = "Die ISBN von Elasticsearch in Action lautet 9781617291623";
041
042        String[] expected = new String[] {
043                "Die",
044                "ISBN",
045                "von",
046                "Elasticsearch",
047                "in",
048                "Action",
049                "lautet",
050                "9781617291623",
051                "EAN 9781617291623",
052                "GTIN 9781617291623",
053                "ISBN 978-1-61729-162-3",
054                "ISBN 9781617291623"
055        };
056        assertSimpleTSOutput(analyzer.tokenStream("content", source), expected);
057    }
058
059    @Test
060    public void testISBN2() throws IOException {
061        String source = "Die ISBN von Lucene in Action lautet 1-9339-8817-7.";
062
063        String[] expected = new String[] {
064                "Die",
065                "ISBN",
066                "von",
067                "Lucene",
068                "in",
069                "Action",
070                "lautet",
071                "1-9339-8817-7.",
072                "ISBN 1-933988-17-7",
073                "ISBN 1933988177",
074                "ISBN 978-1-933988-17-7",
075                "ISBN 9781933988177"
076        };
077        assertSimpleTSOutput(analyzer.tokenStream("content", source), expected);
078    }
079
080    private AnalysisService createAnalysisService() {
081        Settings settings = ImmutableSettings.settingsBuilder()
082                .loadFromClasspath("org/xbib/elasticsearch/index/analysis/standardnumber.json").build();
083
084        Index index = new Index("test");
085
086        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
087                new EnvironmentModule(new Environment(settings)),
088                new IndicesAnalysisModule())
089                .createInjector();
090
091        AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
092        new AnalysisStandardNumberPlugin().onModule(analysisModule);
093
094        Injector injector = new ModulesBuilder().add(
095                new IndexSettingsModule(index, settings),
096                new IndexNameModule(index),
097                analysisModule)
098                .createChildInjector(parentInjector);
099
100        return injector.getInstance(AnalysisService.class);
101    }
102
103    private void assertSimpleTSOutput(TokenStream stream, String[] expected) throws IOException {
104        stream.reset();
105        CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
106        assertNotNull(termAttr);
107        int i = 0;
108        while (stream.incrementToken()) {
109            assertTrue(i < expected.length);
110            assertEquals(termAttr.toString(), expected[i++], "expected different term at index " + i);
111        }
112        assertEquals(i, expected.length, "not all tokens produced");
113        stream.close();
114    }
115}
