001package org.xbib.standardnumber;
002
003import java.util.Arrays;
004import java.util.Collection;
005
006public abstract class AbstractStandardNumber implements StandardNumber {
007
008    public Collection<String> getTypedVariants() {
009        return Arrays.asList(
010            type().toUpperCase() + " " + format(),
011            type().toUpperCase() + " " + normalizedValue());
012
013    }
014}
