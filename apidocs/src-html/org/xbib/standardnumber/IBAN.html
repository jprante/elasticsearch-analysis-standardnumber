001    package org.xbib.standardnumber;
002    
003    import org.xbib.standardnumber.check.iso7064.MOD9710;
004    
005    import java.util.Arrays;
006    import java.util.regex.Matcher;
007    import java.util.regex.Pattern;
008    
009    /**
010     *  ISO 13616 International Bank Account Number (IBAN)
011     *
012     * The International Bank Account Number (IBAN) is an internationally agreed means of
013     * identifying bank accounts across national borders with a reduced risk of transcription
014     * errors. It was originally adopted by the European Committee for Banking Standards (ECBS)
015     * and later as an international standard under ISO 13616:1997. The current standard
016     * is ISO 13616:2007, which indicates SWIFT as the formal registrar.
017     *
018     *  Checksum in accordance to ISO 7064 MOD-97
019     */
020    public class IBAN implements Comparable<IBAN>, StandardNumber {
021    
022        private static final Pattern PATTERN = Pattern.compile("[\\p{Alnum}\\-\\s]{5,34}");
023    
024        private String formatted;
025    
026        private String value;
027    
028        private String country;
029    
030        private boolean createWithChecksum;
031    
032        @Override
033        public int compareTo(IBAN iban) {
034            return iban != null ? normalizedValue().compareTo(iban.normalizedValue()) : -1;
035        }
036    
037        @Override
038        public IBAN set(CharSequence value) {
039            this.value = value != null ? value.toString() : null;
040            return this;
041        }
042    
043        @Override
044        public IBAN checksum() {
045            this.createWithChecksum = true;
046            return this;
047        }
048    
049        @Override
050        public IBAN normalize() {
051            Matcher m = PATTERN.matcher(value);
052            if (m.find()) {
053                this.value = parse(value.substring(m.start(), m.end()));
054            }
055            return this;
056        }
057    
058        @Override
059        public IBAN verify() throws NumberFormatException {
060            check();
061            if (formatted.length() != getLengthForCountryCode(country)) {
062                throw new NumberFormatException("invalid length for country: "
063                        + formatted.length() + " " + formatted);
064            }
065            return this;
066        }
067    
068        @Override
069        public String normalizedValue() {
070            return formatted;
071        }
072    
073        @Override
074        public String format() {
075            return formatted;
076        }
077    
078        private final static MOD9710 check = new MOD9710();
079    
080        private void check() throws NumberFormatException {
081            if (createWithChecksum) {
082                int c = check.compute(value.substring(0, value.length()-2));
083                String chk = String.format("%02d", c);
084                this.value = value + chk;
085                this.formatted = formatted.substring(0,2) + chk + formatted.substring(4);
086            }
087            try {
088                check.verify(value);
089            } catch (NumberFormatException e) {
090                throw new NumberFormatException(e.getMessage() + ": " + formatted);
091            }
092        }
093    
094        private String parse(String raw) {
095            StringBuilder sb = new StringBuilder(raw);
096            int i = sb.indexOf("-");
097            while (i >= 0) {
098                sb.deleteCharAt(i);
099                i = sb.indexOf("-");
100            }
101            i = sb.indexOf(" ");
102            while (i >= 0) {
103                sb.deleteCharAt(i);
104                i = sb.indexOf(" ");
105            }
106            this.formatted = sb.toString();
107            this.country = sb.substring(0,2);
108            // move first 4 characters to last
109            sb = new StringBuilder(sb.substring(4)).append(sb.substring(0,4));
110            // replace characters with decimal values
111            for (i = 0; i < sb.length(); i++) {
112                char ch = sb.charAt(i);
113                if (ch >= 'A' && ch <= 'Z') {
114                    sb.deleteCharAt(i);
115                    String s = Integer.toString(ch - 'A' + 10);
116                    sb.insert(i, s);
117                }
118            }
119            return sb.toString();
120        }

122        /**
123         * Known country codes, this list must be sorted to allow binary search.
124         */
125        private static final String[] COUNTRY_CODES = {
126                "AD", "AE", "AL", "AO", "AT", "AZ", "BA", "BE", "BF", "BG", "BH", "BI", "BJ", "BR", "CG", "CH", "CI",
127                "CM", "CR", "CV", "CY", "CZ", "DE", "DK", "DO", "DZ", "EE", "EG", "ES", "FI", "FO", "FR", "GA", "GB",
128                "GE", "GI", "GL", "GR", "GT", "HR", "HU", "IE", "IL", "IR", "IS", "IT", "KW", "KZ", "LB", "LI", "LT",
129                "LU", "LV", "MC", "MD", "ME", "MG", "MK", "ML", "MR", "MT", "MU", "MZ", "NL", "NO", "PK", "PL", "PS",
130                "PT", "RO", "RS", "SA", "SE", "SI", "SK", "SM", "SN", "TN", "TR", "UA", "VG" };
131        /**
132         * Lengths for each country's IBAN. The indices match the indices of {@link #COUNTRY_CODES}, the values are the expected length.
133         */
134        private static final int[] COUNTRY_IBAN_LENGTHS = {

138                28 /* DO */, 24 /* DZ */, 20 /* EE */, 27 /* EG */, 24 /* ES */, 18 /* FI */, 18 /* FO */, 27 /* FR */,
139                27 /* GA */, 22 /* GB */, 22 /* GE */, 23 /* GI */, 18 /* GL */, 27 /* GR */, 28 /* GT */, 21 /* HR */,
140                28 /* HU */, 22 /* IE */, 23 /* IL */, 26 /* IR */, 26 /* IS */, 27 /* IT */, 30 /* KW */, 20 /* KZ */,
141                28 /* LB */, 21 /* LI */, 20 /* LT */, 20 /* LU */, 21 /* LV */, 27 /* MC */, 24 /* MD */, 22 /* ME */,

146    
147        /**
148         * Returns the IBAN length for a given country code.
149         * @param countryCode a non-null, uppercase, two-character country code.
150         * @return the IBAN length for the given country, or -1 if the input is not a known, two-character country code.
151         * @throws NullPointerException if the input is null.
152         */
153        private int getLengthForCountryCode(String countryCode) {
154            int index = Arrays.binarySearch(COUNTRY_CODES, countryCode);
155            if (index > -1) {
156                return COUNTRY_IBAN_LENGTHS[index];
157            }
158            return -1;
159        }
