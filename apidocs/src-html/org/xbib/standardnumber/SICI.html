001    package org.xbib.standardnumber;
002    
003    import java.util.regex.Matcher;
004    import java.util.regex.Pattern;
005    
006    /**
007     *  Z39.56 Serial Item and Contribution Identifier
008     *
009     * The SICI code (Serial Item and Contribution Identifier) is described in the
010     * American standard ANSI/NISO Z39.56. The SICI code is known among
011     * international scientific publishers and reproduction rights agencies.
012     * The SICI even provides for the unambiguous identification of each article
013     * or contribution published in a given issue of a serial publication.
014     *
015     * The SICI contains:
016     *
017     * - the ISSN
018     *
019     * - the date of publication, between brackets and formatted according to the
020     *   formula YYYYMM
021     *
022     * - the issue number
023     *
024     * - the version number of the standard, here 1, preceded by a semicolon
025     *
026     * - and lastly a hyphen which precedes the control character calculated
027     *   on the basis of all the preceding characters
028     *
029     *  Example:
030     *  0095-4403(199502/03)21:3<12:WATIIB>2.0.TX;2-J
031     *
032     */
033    public class SICI implements Comparable<SICI>, StandardNumber {
034    
035        private static final Pattern PATTERN = Pattern.compile("[\\p{Graph}\\p{Punct}]{12,64}");
036    
037        private String value;
038    
039        private String formatted;
040    
041        private boolean createWithChecksum;
042    
043        @Override
044        public int compareTo(SICI sici) {
045            return sici != null ? normalizedValue().compareTo(sici.normalizedValue()) : -1;
046        }
047    
048        @Override
049        public SICI set(CharSequence value) {
050            this.value = value != null ? value.toString() : null;
051            return this;
052        }
053    
054        @Override
055        public SICI checksum() {
056            this.createWithChecksum = true;
057            return this;
058        }
059    
060        @Override
061        public SICI normalize() {
062            Matcher m = PATTERN.matcher(value);
063            if (m.find()) {
064                this.value = clean(value.substring(m.start(), m.end()));
065            }
066            return this;
067        }
068    
069        @Override
070        public SICI verify() throws NumberFormatException {
071            check();
072            return this;
073        }
074    
075        @Override
076        public String normalizedValue() {
077            return value;
078        }
079    
080        @Override
081        public String format() {
082            return formatted;
083        }
084    
085        private final static String ALPHABET = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ#";
086    
087        private final static int modulus = ALPHABET.length();
088    
089        private void check() throws NumberFormatException {
090            if (value == null) {
091                throw new NumberFormatException("null");
092            }
093            int l = value.length() - 1;
094            int val;
095            int sum = 0;
096            for (int i = 0; i < l; i++) {
097                val = ALPHABET.indexOf(value.charAt(i));
098                sum += val *  (i %2 == 0 ? 1 : 3);
099            }
100            int chk = modulus - sum % modulus;
101            if (createWithChecksum) {
102                char ch = chk > 35 ? '#' : chk > 9 ? (char)(10 + (chk - 'A')) : (char)('0' + chk);
103                value = value.substring(0, l) + ch;
104            }
105            char digit = value.charAt(l);
106            int chk2 = digit == '#' ? 36 : (digit >= '0' && digit <= '9') ? digit - '0' : digit -'A' + 10;
107            boolean valid = chk == chk2;
108            if (!valid) {
109                throw new NumberFormatException("invalid checksum: " + chk + " != " + chk2);
110            }
111        }
112    
113        private String clean(String raw) {
114            if (raw == null) {
115                return raw;
116            }
117            StringBuilder sb = new StringBuilder(raw);
118            int pos = sb.indexOf("SICI ");
119            if (pos >= 0) {
120                sb = new StringBuilder(sb.substring(pos + 5));
121            }
122            this.formatted = "SICI " + sb;
123            return sb.toString();
124        }
125    }
