001    package org.xbib.standardnumber.check;
002    
003    public class LuhnMOD10 implements Digit {
004    
005        public String encode(String digits) {
006            return digits + compute(digits);
007        }
008    
009        public void verify(String digits) throws NumberFormatException {
010            boolean b = (computeSum(digits) % 10) == 0;
011            if (!b) {
012                throw new NumberFormatException("bad checksum");
013            }
014        }
015    
016        public int compute(String digits) {
017            int val = computeSum(digits);
018            return (val == 0) ? 0 : (10 - val);
019        }
020    
021        public int getDigit(String digits) {
022            return digits.charAt(digits.length() - 1) - '0';
023        }
024    
025        public String getNumber(String digits) {
026            return digits.substring(0, digits.length() - 1);
027        }
028    
029        private int computeSum(String digits) {
030            int val = 0;
031            for (int i = 0; i < digits.length(); i += 2) {
032                int c = digits.charAt(i) - '0';
033                if (c < 0 || c > 9) {
034                    throw new NumberFormatException("Bad digit: '" + digits.charAt(i) + "'");
035                }
036                val += c;
037            }
038    
039            for (int i = 1; i < digits.length(); i += 2) {
040                int c = digits.charAt(i) - '0';
041                if (c < 0 || c > 9) {
042                    throw new NumberFormatException("Bad digit: '" + digits.charAt(i) + "'");
043                }
044                val += (c >= 5) ? (2*c - 9) : (2*c);
045            }
046            return val % 10;
047        }
048    }
