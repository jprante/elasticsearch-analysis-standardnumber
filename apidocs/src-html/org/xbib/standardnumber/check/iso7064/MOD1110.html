001    package org.xbib.standardnumber.check.iso7064;
002    
003    import org.xbib.standardnumber.check.Digit;
004    
005    public class MOD1110 implements Digit {
006    
007        private final static String ALPHABET = "0123456789";
008    
009        @Override
010        public String encode(String digits) {
011            int c = compute(digits);
012            return digits + c;
013        }
014    
015        @Override
016        public void verify(String digits) throws NumberFormatException {
017            boolean b = compute(digits) == 1;
018            if (!b) {
019                throw new NumberFormatException("bad checksum: " + digits);
020            }
021        }
022    
023        @Override
024        public int compute(String digits) {
025            int modulus = ALPHABET.length();
026            int check = modulus / 2;
027            for (int i = 0; i < digits.length(); i++) {
028                check = (((check > 0 ? check : modulus) * 2) % (modulus + 1)
029                        + ALPHABET.indexOf(digits.charAt(i))) % modulus;
030            }
031            return check;
032        }
033    
034        @Override
035        public int getDigit(String digits) {
036            return digits.charAt(digits.length() - 1);
037        }
038    
039        @Override
040        public String getNumber(String digits) {
041            return digits.substring(0, digits.length() - 1);
042        }
043    }
