001    package org.xbib.standardnumber.check.iso7064;
002    
003    import org.xbib.standardnumber.check.Digit;
004    
005    public class MOD112 implements Digit {
006    
007        @Override
008        public String encode(String digits) {
009            int c = compute(digits);
010            return digits + (c == 10 ? 'X' : c);
011        }
012    
013        @Override
014        public void verify(String digits) throws NumberFormatException {
015            boolean b = compute(getNumber(digits)) == getDigit(digits);
016            if (!b) {
017                throw new NumberFormatException("bad checksum: " + digits);
018            }
019        }
020    
021        @Override
022        public int compute(String digits) {
023            int p = 0;
024            for (int i = 0; i < digits.length(); ++i) {
025                int c = digits.charAt(i) - '0';
026                if (c < 0 || c > 9) {
027                    throw new NumberFormatException("'" + digits + "' has bad digit: '" + digits.charAt(i) + "'");
028                }
029                p = 2*(p + c);
030            }
031            p = p % 11;
032            return (12 - p) % 11;
033        }
034    
035        @Override
036        public int getDigit(String digits) {
037            char c = digits.charAt(digits.length() -1);
038            return  c == 'X' || c == 'x' ? 10 : c - '0';
039        }
040    
041        @Override
042        public String getNumber(String digits) {
043            return digits.substring(0, digits.length() -1);
044        }
045    
046    }
