001    package org.xbib.standardnumber;
002    
003    import java.util.regex.Matcher;
004    import java.util.regex.Pattern;
005    
006    /**
007     * ISO 21047: International Standard Text Code (ISTC)
008     *
009     * The International Standard Text Code (ISTC) is a numbering system for the unique identification
010     * of text-based works; the term “work” can refer to any content appearing in conventional
011     * printed books, audio-books, static e-books or enhanced digital books, as well as content
012     * which might appear in a newspaper or journal.
013     *
014     * The ISTC provides sales analysis systems, retail websites, library catalogs and other
015     * bibliographic systems with a method of automatically linking together publications
016     * of the “same content” and/or “related content”, thus improving discoverability of
017     * products and efficiencies.
018     *
019     * An ISTC number is the link between a user’s search for a piece of content and the
020     * ultimate sale or loan of a publication.
021     *
022     * The standard was formally published in March 2009
023     *
024     * Checksum algorithm is ISO 7064 MOD 16/3
025     *
026     */
027    public class ISTC implements Comparable<ISTC>, StandardNumber {
028    
029        private static final Pattern PATTERN = Pattern.compile("[\\p{Alnum}\\-\\s]{12,24}");
030    
031        private String value;
032    
033        private String formatted;
034    
035        private boolean createWithChecksum;
036    
037        @Override
038        public int compareTo(ISTC istc) {
039            return istc != null ? normalizedValue().compareTo(istc.normalizedValue()) : -1;
040        }
041    
042        @Override
043        public ISTC set(CharSequence value) {
044            this.value = value != null ? value.toString() : null;
045            return this;
046        }
047    
048        @Override
049        public ISTC checksum() {
050            this.createWithChecksum = true;
051            return this;
052        }
053    
054        @Override
055        public ISTC normalize() {
056            Matcher m = PATTERN.matcher(value);
057            if (m.find()) {
058                this.value = clean(value.substring(m.start(), m.end()));
059            }
060            return this;
061        }
062    
063        @Override
064        public ISTC verify() throws NumberFormatException {
065            check();
066            return this;
067        }
068    
069        @Override
070        public String normalizedValue() {
071            return value;
072        }
073    
074        @Override
075        public String format() {
076            return formatted;
077        }
078    
079        private void check() throws NumberFormatException {
080            if (value == null) {
081                throw new NumberFormatException("null");
082            }
083            int l = value.length() - 1;
084            int checksum = 0;
085            int weight;
086            int factor;
087            int val;
088            for (int i = 0; i < l; i++) {
089                val = value.charAt(i);
090                if (val >= 'A' && val <= 'Z') {
091                    val = 10 + (val - 'A');
092                }
093                if (val >= '0' && val <= '9') {
094                    val = val - '0';
095                }
096                if (val < 0 || val > 35) {
097                    throw new NumberFormatException("not a valid symbol: " + val );
098                }
099                factor = i % 4 < 2 ? 1 : 5;
100                weight = (12 - 2 * (i % 4)) - factor; // --> 11,9,3,1
101                checksum += val * weight;
102            }
103            int chk = checksum % 16;
104            if (createWithChecksum) {
105                char ch = chk > 9 ? (char)(10 + (chk - 'A')) : (char)('0' + chk);
106                value = value.substring(0, l) + ch;
107            }
108            char digit = value.charAt(l);
109            int chk2 = (digit >= '0' && digit <= '9') ? digit - '0' : digit -'A' + 10;
110            boolean valid = chk == chk2;
111            if (!valid) {
112                throw new NumberFormatException("invalid checksum: " + chk + " != " + chk2);
113            }
114        }
115    
116        private String clean(String raw) {
117            if (raw == null) {
118                return raw;
119            }
120            StringBuilder sb = new StringBuilder(raw);
121            int i = sb.indexOf("-");
122            while (i >= 0) {
123                sb.deleteCharAt(i);
124                i = sb.indexOf("-");
125            }
126            i = sb.indexOf(" ");
127            while (i >= 0) {
128                sb.deleteCharAt(i);
129                i = sb.indexOf(" ");
130            }
131            if (sb.indexOf("ISTC") == 0) {
132                sb = new StringBuilder(sb.substring(4));
133            }
134            this.formatted = "ISTC "
135                    + sb.substring(0,3) + "-"
136                    + sb.substring(3,7) + "-"
137                    + sb.substring(7,15) + "-"
138                    + sb.substring(15);
139            return sb.toString();
140        }
141    }
