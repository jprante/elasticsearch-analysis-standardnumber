001    package org.xbib.standardnumber;
002    
003    import java.net.URI;
004    import java.util.regex.Matcher;
005    import java.util.regex.Pattern;
006    
007    /**
008     * ARK Archival Resource Key
009     *
010     * An ARK is a Uniform Resource Locator (URL) that is a multi-purpose identifier
011     * for information objects of any type. An ARK contains the label ark: after the
012     * hostname, an URL request terminated by '?' returns a brief metadata record,
013     * and an URL request terminated by '??' returns metadata that includes a commitment
014     * statement from the current service provider.
015     *
016     * The ARK and its inflections ('?' and '??') gain access to three facets of a
017     * provider's ability to provide persistence.
018     *
019     * Implicit in the design of the ARK scheme is that persistence is purely a matter
020     * of service and not a property of a naming syntax.
021     *
022     * @see <a href="http://tools.ietf.org/html/draft-kunze-ark-18">ARK IETF RFC</a>
023     * 
024     * @see <a href="http://www.cdlib.org/services/uc3/docs/jak_ARKs_Berlin_2012.pdf">10 years ARK</a>
025     */
026    public class ARK implements Comparable<ARK>, StandardNumber {
027    
028        private static final Pattern PATTERN = Pattern.compile("[\\p{Graph}\\p{Punct}]{0,48}");
029    
030        private static final Pattern URI_PATTERN = Pattern.compile("^(ark)://[-a-zA-Z0-9+&@#/%?=~_|!:,.;]*[-a-zA-Z0-9+&@#/%=~_|]");
031    
032        private Object value;
033    
034        @Override
035        public int compareTo(ARK ark) {
036            return ark != null ? normalizedValue().compareTo(ark.normalizedValue()) : -1;
037        }
038    
039        @Override
040        public ARK set(CharSequence value) {
041            this.value = value != null ? value.toString() : null;
042            return this;
043        }
044    
045        /**
046         * No checksum
047         * @return this ARK
048         */
049        @Override
050        public ARK checksum() {
051            return this;
052        }
053    
054        @Override
055        public ARK normalize() {
056            if (value == null) {
057                return this;
058            }
059            String s = value.toString();
060            Matcher m = URI_PATTERN.matcher(s);
061            if (m.find()) {
062                this.value = URI.create(s.substring(m.start(), m.end()));
063            }
064            m = PATTERN.matcher(s);
065            if (m.find()) {
066                this.value = URI.create(s.substring(m.start(), m.end()));
067            }
068            return this;
069        }
070    
071        /**
072         * No verification.
073         *
074         * @return this ARK
075         * @throws NumberFormatException
076         */
077        @Override
078        public ARK verify() throws NumberFormatException {
079            return this;
080        }
081    
082        @Override
083        public String normalizedValue() {
084            return value != null ? value.toString() : null;
085        }
086    
087        @Override
088        public String format() {
089            return value != null ? value.toString() : null;
090        }
091    
092        public URI asURI() {
093            return (URI)value;
094        }
095    }
