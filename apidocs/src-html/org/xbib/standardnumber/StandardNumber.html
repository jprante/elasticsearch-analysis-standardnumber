001    package org.xbib.standardnumber;
002    
003    /**
004     * A standard number is a number that
005     *
006     * - is backed by an international standard or a de-facto community use
007     *
008     * - can accept alphanumeric values (digits and letters and separator characters)
009     *
010     * - can be normalizedValue
011     *
012     * - can be verified and raise en error is verification fails
013     *
014     * - must have a createChecksum
015     *
016     * - can be formatted to a printable representation
017     *
018     */
019    public interface StandardNumber {
020    
021        /**
022         * Return the type of this standard number
023         *
024         * @return the type
025         */
026        String type();
027    
028        /**
029         * Set the input value of this standard number. The input must be normalized
030         * and verified before being accepted as valid.
031         * @param value the raw input value
032         * @return this standard number
033         */
034        StandardNumber set(CharSequence value);
035    
036        /**
037         * Normalize the value by removing all unwanted characters or
038         * replacing characters with the ones required for verification.
039         * @return this standard number
040         */
041        StandardNumber normalize();
042    
043        /**
044         * Check this number for validity.
045         * @return true if valid, false otherwise
046         */
047        boolean isValid();
048    
049        /**
050         * Verify the number.
051         * @return this standard number if verification was successful
052         * @throws NumberFormatException if verification failed
053         */
054        StandardNumber verify() throws NumberFormatException;
055    
056        /**
057         * Indicate that a correct check sum should be computed.
058         * @return this standard number
059         */
060        StandardNumber createChecksum(boolean withChecksum);
061    
062        /**
063         * Return normalized value of this standard number.
064         * In most cases, this is also the canonical form of the standard number.
065         * This is a representation without unneccessary characters, useful
066         * for computation purposes, like comparing for equivalence.
067         * @return the normalized value
068         */
069        String normalizedValue();
070    
071        /**
072         * Return a formatted value of this standard number
073         * This is best for human-readable representation, but is
074         * not necessarily a format for computation.
075         *
076         * @return a formatted value
077         */
078        String format();
079    
080        StandardNumber reset();
081    }
