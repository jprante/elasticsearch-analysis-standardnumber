002    package org.xbib.standardnumber;
003    
004    import java.util.regex.Matcher;
005    import java.util.regex.Pattern;
006    
007    /**
008     * Zeitschriftendatenbank ID
009     *
010     *  ZDB is the world’s largest specialized database for serial titles (journals, annuals, newspapers, also
011     *  e-journals).
012     *
013     * @see <a href="http://support.d-nb.de/iltis/onlineRoutinen/Pruefziffernberechnung.htm"></a>Prüfziffernberechnung in ILTIS</a>
014     * @see <a href="https://wiki.dnb.de/pages/viewpage.action?pageId=48139522">DNB Wiki</a>
015     */
016    public class ZDB implements Comparable<ZDB>, StandardNumber {
017    
018        private final static Pattern PATTERN = Pattern.compile("[\\p{Digit}xX\\-]{0,11}");
019    
020        private String value;
021    
022        private String formatted;
023    
024        private boolean createWithChecksum;
025    
026        @Override
027        public ZDB set(CharSequence value) {
028            this.value = value != null ? value.toString() : null;
029            return this;
030        }
031    
032        @Override
033        public int compareTo(ZDB o) {
034            return value != null ?  value.compareTo(o.normalizedValue()) : -1;
035        }
036    
037        @Override
038        public String normalizedValue() {
039            return value;
040        }
041    
042        @Override
043        public ZDB checksum() {
044            this.createWithChecksum = true;
045            return this;
046        }
047    
048        @Override
049        public ZDB normalize() {
050            Matcher m = PATTERN.matcher(value);
051            if (m.find()) {
052                this.value = dehyphenate(value.substring(m.start(), m.end()));
053            }
054            return this;
055        }
056    
057        @Override
058        public ZDB verify() throws NumberFormatException {
059            check();
060            return this;
061        }
062    
063        @Override
064        public String format() {
065            if (formatted == null) {
066                StringBuilder sb = new StringBuilder(value);
067                this.formatted = sb.insert(sb.length()-1,"-").toString();
068            }
069            return formatted;
070        }
071    
072        private void check() throws NumberFormatException {
073            int l = value.length() - 1;
074            int checksum = 0;
075            int weight = 2;
076            int val;
077            for (int i = l-1; i >= 0; i--) {
078                val = value.charAt(i) - '0';
079                if (val < 0 || val > 9) {
080                    throw new NumberFormatException("not a digit: " + val );
081                }
082                checksum += val * weight++;
083            }
084            if (createWithChecksum) {
085                char ch = checksum % 11 == 10 ? 'X' : (char)('0' + (checksum % 11));
086                value = value.substring(0, l) + ch;
087            }
088            boolean valid = checksum % 11 ==
089                    (value.charAt(l) == 'X' || value.charAt(l) == 'x' ? 10 : value.charAt(l) - '0');
090            if (!valid) {
091                throw new NumberFormatException("invalid checksum: " + value.charAt(l));
092            }
093        }
094    
095        private String dehyphenate(String value) {
096            StringBuilder sb = new StringBuilder(value);
097            int i = sb.indexOf("-");
098            while (i >= 0) {
099                sb.deleteCharAt(i);
100                i = sb.indexOf("-");
101            }
102            return sb.toString();
103        }
104    }
