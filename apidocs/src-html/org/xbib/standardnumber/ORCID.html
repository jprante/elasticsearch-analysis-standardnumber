001    package org.xbib.standardnumber;
002    
003    import java.net.URI;
004    
005    /**
006     * ORCID
007     *
008     * ORCID is comptabible to International Standard Name Identifier (ISNI)  ISO 2772
009     *
010     * Checksum is on accordance to ISO/IEC 7064:2003, MOD 11-2
011     */
012    public class ORCID extends ISNI {
013    
014        @Override
015        public ORCID set(CharSequence value) {
016            super.set(value);
017            return this;
018        }
019    
020        @Override
021        public ORCID checksum() {
022            super.checksum();
023            return this;
024        }
025    
026        @Override
027        public ORCID normalize() {
028            super.normalize();
029            return this;
030        }
031    
032        @Override
033        public ORCID verify() throws NumberFormatException {
034            super.verify();
035            return this;
036        }
037    
038        public URI toURI() {
039            return URI.create("http://orcid.org/" + normalizedValue());
040        }
041    
042    }
