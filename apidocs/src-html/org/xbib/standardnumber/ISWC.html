001    package org.xbib.standardnumber;
002    
003    import java.util.regex.Matcher;
004    import java.util.regex.Pattern;
005    
006    /**
007     * ISO 15707: International Standard Musical Work Code (ISWC)
008     *
009     * International Standard Musical Work Code (ISWC) is a unique identifier for
010     * musical works, similar to ISBN.
011     *
012     * Its primary purpose is in collecting society administration, and to clearly identify works in
013     * legal contracts. It would also be useful in library cataloging.
014     *
015     * Due to the fact that a musical work can have multiple authors, it is inevitable that,
016     * on rare occasions, a duplicate ISWC might exist and might not be detected immediately.
017     *
018     * Because of the existing business practices among collecting societies, it is not possible
019     * to simply declare an ISWC as obsolete. In such cases, as soon as they are identified,
020     * the system will deal with duplicate registrations by linking such registration records
021     * in the ISWC database.
022     *
023     */
024    public class ISWC implements Comparable<ISWC>, StandardNumber {
025    
026        private static final Pattern PATTERN = Pattern.compile("[\\p{Alnum}\\-\\s]{10,13}");
027    
028        private String value;
029    
030        private String formatted;
031    
032        @Override
033        public int compareTo(ISWC iswc) {
034            return iswc != null ? normalizedValue().compareTo(iswc.normalizedValue()) : -1;
035        }
036    
037        @Override
038        public ISWC set(CharSequence value) {
039            this.value = value != null ? value.toString() : null;
040            return this;
041        }
042    
043        @Override
044        public ISWC checksum() {
045            return this;
046        }
047    
048        @Override
049        public ISWC normalize() {
050            Matcher m = PATTERN.matcher(value);
051            if (m.find()) {
052                this.value = clean(value.substring(m.start(), m.end()));
053            }
054            return this;
055        }
056    
057        @Override
058        public ISWC verify() throws NumberFormatException {
059            check();
060            return this;
061        }
062    
063        @Override
064        public String normalizedValue() {
065            return value;
066        }
067    
068        @Override
069        public String format() {
070            return formatted;
071        }
072    
073        private void check() throws NumberFormatException {
074            if (value == null) {
075                throw new NumberFormatException("null");
076            }
077            int l = value.length();
078            int checksum = 1;
079            int val;
080            int weight;
081            for (int i = 1; i < l; i++) {
082                val = value.charAt(i) - '0';
083                if (val < 0 || val > 9) {
084                    throw new NumberFormatException("not a digit: " + val );
085                }
086                weight = i < l - 1 ? i : 1;
087                checksum += val * weight;
088            }
089            int chk = checksum % 10;
090            boolean valid = chk == 0;
091            if (!valid) {
092                throw new NumberFormatException("invalid checksum: " + chk);
093            }
094        }
095    
096        private String clean(String raw) {
097            if (raw == null) {
098                return raw;
099            }
100            StringBuilder sb = new StringBuilder(raw);
101            int i = sb.indexOf("-");
102            while (i >= 0) {
103                sb.deleteCharAt(i);
104                i = sb.indexOf("-");
105            }
106            i = sb.indexOf(" ");
107            while (i >= 0) {
108                sb.deleteCharAt(i);
109                i = sb.indexOf(" ");
110            }
111            if (sb.indexOf("ISWC") == 0) {
112                sb = new StringBuilder(sb.substring(4));
113            }
114            this.formatted = "ISWC "
115                    + "T-"
116                    + sb.substring(1,10) + "-"
117                    + sb.substring(10,11);
118            return sb.toString();
119        }
120    }
