018    /**
019     * ISO 2108: International Standard Book Number (ISBN)
020     *
021     * Z39.50 BIB-1 Use Attribute 7
022     *
023     * The International Standard Book Number is a 13-digit number
024     * that uniquely identifies books and book-like products published
025     * internationally.
026     *
027     * The purpose of the ISBN is to establish and identify one title or
028     * edition of a title from one specific publisher
029     * and is unique to that edition, allowing for more efficient marketing of products by booksellers,
030     * libraries, universities, wholesalers and distributors.
031     *
032     * Every ISBN consists of thirteen digits and whenever it is printed it is preceded by the letters ISBN.
033     * The thirteen-digit number is divided into four parts of variable length, each part separated by a hyphen.
034     *
035     * This class is based upon the ISBN converter and formatter class
036     * Copyright 2000-2005 by Openly Informatics, Inc. http://www.openly.com/
037     *
038     * @see <a href="http://www.s.org/standards/home/s/international/html/usm12.htm">The ISBN Users' Manual</a>
039     * @see <a href="http://www.ietf.org/html.charters/OLD/urn-charter.html">The IETF URN Charter</a>
040     * @see <a href="http://www.iana.org/assignments/urn-namespaces">The IANA URN assignments</a>
041     * @see <a href="http://www.isbn-international.org/download/List%20of%20Ranges.pdf">ISBN prefix list</a>
042     */
043    public class ISBN implements Comparable<ISBN>, StandardNumber {

059        /**
060         * Set ISBN value
061         *
062         * @param value the ISBN candidate string
063         */
064        @Override
065        public ISBN set(CharSequence value) {
066            this.value = value != null ? value.toString() : null;
067            return this;
068        }
069    
070        @Override
071        public ISBN checksum() {
072            this.createWithChecksum = true;
073            return this;
074        }
075    
076        @Override
077        public int compareTo(ISBN isbn) {
078            return value != null ? value.compareTo(isbn.normalizedValue()): -1;
079        }
080    
081        @Override
082        public ISBN normalize() {
083            Matcher m = PATTERN.matcher(value);
084            this.value = m.find() ? dehyphenate(value.substring(m.start(), m.end())) : null;
085            return this;
086        }

088        /**
089         * Check for this ISBN number validity
090         *
091         * @return true if valid, false otherwise
092         */
093        @Override
094        public ISBN verify() throws NumberFormatException {
095            check();
096            this.valid = eanPreferred ? eanvalue != null : value != null;
097            if (!valid) {
098                throw new NumberFormatException("invalid number");
099            }
100            return this;
101        }

103        /**
104         * Get the normalized value of this standard book number
105         * 
106         * @return the value of this standard book number
107         */
108        @Override
109        public String normalizedValue() {
110            return eanPreferred ? eanvalue : value;
111        }

113        /**
114         * Get printable representation of this standard book number
115         *
116         * @return ISBN-13, with (fixed) check digit
117         */
118        @Override
119        public String format() {
120            return valid ? eanPreferred ?
121                            fix(eanvalue != null ? eanvalue : "978" + value) :
122                            fix("978" +value).substring(4) :
123                    "";
124        }

127            return eanvalue != null;
128        }

134            this.eanPreferred = preferEAN;
135            return this;

138        /**
139         * Get country and publisher code
140         *
141         * @return the country/publisher code from ISBN
142         */
143        public String getCountryAndPublisherCode()  {
144            // we don't care about the wrong checksum when we fix the value
145            String code = eanvalue != null ? fix(eanvalue) : fix("978" + value);
146            String s = code.substring(4);
147            int pos1 = s.indexOf('-');
148            if (pos1 <= 0) {
149                return null;
150            }
151            String pubCode = s.substring(pos1 + 1);
152            int pos2 = pubCode.indexOf('-');
153            if (pos2 <= 0) {
154                return null;
155            }
156            return code.substring(0, pos1 + pos2 + 5);
157        }

159        private String hyphenate(String prefix, String isbn) {
160            StringBuilder sb = new StringBuilder(prefix.substring(0, 4)); // '978-', '979-'
161            prefix = prefix.substring(4);
162            isbn = isbn.substring(3); // 978, 979
163            int i = 0;
164            int j = 0;
165            while (i < prefix.length()) {
166                char ch = prefix.charAt(i++);
167                if (ch == '-') {
168                    sb.append('-'); // set first hyphen
169                } else {
170                    sb.append(isbn.charAt(j++));
171                }
172            }
173            sb.append('-'); // set second hyphen
174            while (j < (isbn.length() - 1)) {
175                sb.append(isbn.charAt(j++));
176            }
177            sb.append('-'); // set third hyphen
178            sb.append(isbn.charAt(isbn.length() - 1));
179            return sb.toString();
180        }

182        private void check() throws NumberFormatException {
183            if (value == null) {
184                throw new NumberFormatException("must not be null");
185            }
186            eanvalue = null;
187            int i;
188            int val;
189            if (value.length() < 9) {
190                throw new NumberFormatException("too short");
191            }
192            if (value.length() == 10) {
193                // ISBN-10
194                int checksum = 0;
195                int weight = 10;
196                for (i = 0; weight > 0; i++) {
197                    val = value.charAt(i) == 'X' || value.charAt(i) == 'x' ? 10
198                            : value.charAt(i) - '0';
199                    if (val >= 0) {
200                        if (val == 10 && weight != 1) {
201                            throw new NumberFormatException("bad X symbol"); // bad place for X
202                        }
203                        checksum += weight * val;
204                        weight--;
205                    } else {
206                        throw new NumberFormatException("invalid char"); // invalid char
207                    }
208                }
209                String s = value.substring(0, 9);
210                if (checksum % 11 != 0) {
211                    if (createWithChecksum) {
212                        this.value = s + createCheckDigit10(s);
213                    } else {
214                        throw new NumberFormatException("bad checksum"); // bad checksum
215                    }
216                }
217                if (eanPreferred) {
218                    this.eanvalue = "978" + s + createCheckDigit13("978" + s);
219                }
220            } else if (value.length() == 13) {
221                // ISBN-13 "book land"
222                if (!value.startsWith("978") && !value.startsWith("979")) {
223                    throw new NumberFormatException("bad prefix, must be 978 or 979: " + value);
224                }
225                int checksum13 = 0;
226                int weight13 = 1;
227                for (i = 0; i < 13; i++) {
228                    val = value.charAt(i) == 'X' || value.charAt(i) == 'x' ? 10 : value.charAt(i) - '0';
229                    if (val >= 0) {
230                        if (val == 10) {
231                            throw new NumberFormatException("bad symbol: " + value);
232                        }
233                        checksum13 += (weight13 * val);
234                        weight13 = (weight13 + 2) % 4;
235                    } else {
236                        throw new NumberFormatException("invalid char");
237                    }
238                }
239                // set value
240                if ((checksum13 % 10) != 0) {
241                    if (eanPreferred && createWithChecksum) {
242                        // with checksum
243                        this.eanvalue = value.substring(0, 12) + createCheckDigit13(value.substring(0, 12));
244                    } else {
245                        throw new NumberFormatException("bad checksum");
246                    }
247                } else {
248                    // correct checksum
249                    this.eanvalue = value;
250                }
251                if (!eanPreferred && (eanvalue.startsWith("978") || eanvalue.startsWith("979"))) {
252                    // create 10-digit from 13-digit
253                    this.value = eanvalue.substring(3, 12) + createCheckDigit10(eanvalue.substring(3, 12));
254                } else {
255                    // 10 digit version not available - not an error
256                    this.value = null;
257                }
258            } else if (value.length() == 9) {
259                String s = value.substring(0, 9);
260                // repair ISBN-10 ?
261                if (createWithChecksum) {
262                    // create 978 from 10-digit without checksum
263                    this.eanvalue = "978" + s + createCheckDigit13("978" + s);
264                    this.value = s + createCheckDigit10(s);
265                } else {
266                    throw new NumberFormatException("invalid");
267                }
268            } else if (value.length() == 12) {
269                // repair ISBN-13 ?
270                if (!value.startsWith("978") && !value.startsWith("979")) {
271                    throw new NumberFormatException("bad prefix");
272                }
273                if (createWithChecksum) {
274                    String s = value.substring(0, 9);
275                    String t = value.substring(3, 12);
276                    // create 978 from 10-digit
277                    this.eanvalue = "978" + s + createCheckDigit13("978" + s);
278                    this.value = t + createCheckDigit10(t);
279                } else {
280                    throw new NumberFormatException("invalid");
281                }
282            } else {
283                throw new NumberFormatException("invalid");
284            }
285        }

287        /**
288         * Returns a ISBN check digit for the first 9 digits in a string
289         *
290         * @param value the value
291         * @return check digit
292         *
293         * @throws NumberFormatException
294         */
295        private char createCheckDigit10(String value) throws NumberFormatException {
296            int checksum = 0;
297            int val;
298            int l = value.length();
299            for (int i = 0; i < l; i++) {
300                val = value.charAt(i) - '0';
301                if (val < 0 || val > 9) {
302                    throw new NumberFormatException("not a digit in " + value);
303                }
304                checksum += val * (10-i);
305            }
306            int mod = checksum % 11;
307            return mod == 0 ? '0' : mod == 1 ? 'X' : (char)((11-mod) + '0');
308        }
309    
310        /**
311         * Returns an ISBN check digit for the first 12 digits in a string
312         *
313         * @param value the value
314         * @return check digit
315         *
316         * @throws NumberFormatException
317         */
318        private char createCheckDigit13(String value) throws NumberFormatException {
319            int checksum = 0;
320            int weight;
321            int val;
322            int l = value.length();
323            for (int i = 0; i < l; i++) {
324                val = value.charAt(i) - '0';
325                if (val < 0 || val > 9) {
326                    throw new NumberFormatException("not a digit in " + value);
327                }
328                weight = i % 2 == 0 ? 1 : 3;
329                checksum += weight * val;
330            }
331            int mod = 10 - checksum % 10;
332            return mod == 10 ? '0' : (char)(mod + '0');
333        }

335        private String fix(String isbn) {
336            if (isbn == null) {
337                return null;
338            }
339            for (int i = 0; i < ranges.size(); i += 2) {
340                if (isInRange(isbn, ranges.get(i), ranges.get(i + 1)) == 0) {
341                    return hyphenate(ranges.get(i), isbn);
342                }
343            }
344            return isbn;
345        }
346    
347        /**
348         * Check if ISBN is within a given value range
349         * @param isbn ISBN to check
350         * @param begin lower ISBN
351         * @param end  higher ISBN
352         * @return -1 if too low, 1 if too high, 0 if range matches
353         */
354        private int isInRange(String isbn, String begin, String end) {
355            String b = dehyphenate(begin);
356            int blen = b.length();
357            int c = blen <= isbn.length() ?
358                    isbn.substring(0, blen).compareTo(b) :
359                    isbn.compareTo(b);
360            if (c < 0) {
361                return -1;
362            }
363            String e = dehyphenate(end);
364            int elen = e.length();
365            c = e.compareTo(isbn.substring(0, elen));
366            if (c < 0) {
367                return 1;
368            }
369            return 0;
370        }
371    
372        private String dehyphenate(String isbn) {
373            StringBuilder sb = new StringBuilder(isbn);
374            int i = sb.indexOf("-");
375            while (i >= 0) {
376                sb.deleteCharAt(i);
377                i = sb.indexOf("-");
378            }
379            return sb.toString();
380        }

382        private final static class ISBNRangeMessageConfigurator {
383    
384            private final Stack<StringBuilder> content;
385    
386            private final List<String> ranges;
387    
388            private String prefix;
389    
390            private String rangeBegin;
391    
392            private String rangeEnd;
393    
394            private int length;
395    
396            private boolean valid;
397    
398            public ISBNRangeMessageConfigurator() {
399                content = new Stack<StringBuilder>();
400                ranges = new ArrayList<String>();
401                length = 0;
402                try {
403                    InputStream in = getClass().getResourceAsStream("/org/xbib/standardnumber/RangeMessage.xml");
404                    XMLInputFactory xmlInputFactory = XMLInputFactory.newInstance();
405                    XMLEventReader xmlReader = xmlInputFactory.createXMLEventReader(in);
406                    while (xmlReader.hasNext()) {
407                        processEvent(xmlReader.peek());
408                        xmlReader.nextEvent();
409                    }
410                } catch (XMLStreamException e) {
411                    throw new RuntimeException(e.getMessage());
412                }
413            }
414    
415            private void processEvent(XMLEvent e) {
416                switch (e.getEventType()) {
417                    case XMLEvent.START_ELEMENT: {
418                        StartElement element = e.asStartElement();
419                        String name = element.getName().getLocalPart();
420                        if ("RegistrationGroups".equals(name)) {
421                            valid = true;
422                        }
423                        content.push(new StringBuilder());
424                        break;
425                    }
426                    case XMLEvent.END_ELEMENT: {
427                        EndElement element = e.asEndElement();
428                        String name = element.getName().getLocalPart();
429                        String v = content.pop().toString();
430                        if ("Prefix".equals(name)) {
431                            prefix = v;
432                        }
433                        if ("Range".equals(name)) {
434                            int pos = v.indexOf('-');
435                            if (pos > 0) {
436                                rangeBegin = v.substring(0, pos);
437                                rangeEnd = v.substring(pos + 1);
438                            }
439                        }
440                        if ("Length".equals(name)) {
441                            length = Integer.parseInt(v);
442                        }
443                        if ("Rule".equals(name)) {
444                            if (valid && rangeBegin != null && rangeEnd != null) {
445                                if (length > 0) {
446                                    ranges.add(prefix + "-" + rangeBegin.substring(0, length));
447                                    ranges.add(prefix + "-" + rangeEnd.substring(0, length));
448                                }
449                            }
450                        }
451                        break;
452                    }
453                    case XMLEvent.CHARACTERS: {
454                        Characters c = (Characters) e;
455                        if (!c.isIgnorableWhiteSpace()) {
456                            String text = c.getData().trim();
457                            if (text.length() > 0 && !content.empty()) {
458                                content.peek().append(text);
459                            }
460                        }
461                        break;
462                    }
463                }
464            }
465    
466            public List<String> getRanges() {
467                return ranges;
468            }
469        }
