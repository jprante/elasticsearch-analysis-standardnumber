001    package org.xbib.standardnumber;
002    
003    import java.net.URI;
004    import java.util.regex.Matcher;
005    import java.util.regex.Pattern;
006    
007    /**
008     * ISO 26324: Digital Object Identifier System (DOI)
009     *
010     * Z39.50 BIB-1 Use Attribute 1094
011     *
012     * DOI is an acronym for "digital object identifier", meaning a "digital identifier of an object"
013     * rather than an "identifier of a digital object". The DOI system was initiated by the
014     * International DOI Foundation in 1998, and initially developed with the collaboration
015     * of some participants in ISO/TC46/SC9. Due to its application in the fields of
016     * information and documentation and previous collaboration with some ISO/TC46/SC9 participants,
017     * it was introduced as a possible work item in 2004 and further developed from 2006 to 2010.
018     *
019     * The DOI system is designed to work over the Internet. A DOI name is permanently assigned
020     * to an object to provide a resolvable persistent network link to current information about
021     * that object, including where the object, or information about it, can be found on the
022     * Internet. While information about an object can change over time, its DOI name will not
023     * change. A DOI name can be resolved within the DOI system to values of one or more types
024     * of data relating to the object identified by that DOI name, such as a URL, an e-mail address,
025     * other identifiers and descriptive metadata.
026     *
027     * The DOI system enables the construction of automated services and transactions.
028     * Applications of the DOI system include but are not limited to managing information
029     * and documentation location and access; managing metadata; facilitating electronic
030     * transactions; persistent unique identification of any form of any data; and commercial
031     * and non-commercial transactions.
032     *
033     * The content of an object associated with a DOI name is described unambiguously
034     * by DOI metadata, based on a structured extensible data model that enables the object
035     * to be associated with metadata of any desired degree of precision and granularity
036     * to support description and services. The data model supports interoperability
037     * between DOI applications.
038     *
039     * The scope of the DOI system is not defined by reference to the type of content
040     * (format, etc.) of the referent, but by reference to the functionalities it provides
041     * and the context of use. The DOI system provides, within networks of DOI applications,
042     * for unique identification, persistence, resolution, metadata and semantic interoperability.
043     *
044     */
045    public class DOI implements Comparable<DOI>, StandardNumber {
046    
047        private static final Pattern PATTERN = Pattern.compile("\\b10\\.\\d{4}([.][0-9]+)*/[/-a-z0-9.()<>_:;\\\\]+\\b");
048    
049        private static final Pattern URI_PATTERN = Pattern.compile("\\b(https?|ftp|file)://[-a-zA-Z0-9+&@#/%?=~_|!:,.;]*[-a-zA-Z0-9+&@#/%=~_|]\\b");
050    
051        private String prefix = "http://doi.org/";
052    
053        private Object raw;
054    
055        private URI value;
056    
057        @Override
058        public String type() {
059            return "doi";
060        }
061    
062        @Override
063        public int compareTo(DOI doi) {
064            return doi != null ? normalizedValue().compareTo(doi.normalizedValue()) : -1;
065        }
066    
067        @Override
068        public DOI set(CharSequence value) {
069            this.raw = value;
070            return this;
071        }
072    
073        @Override
074        public DOI createChecksum(boolean checksum) {
075            return this;
076        }
077    
078        @Override
079        public DOI normalize() {
080            if (raw == null) {
081                return this;
082            }
083            this.value = make(raw.toString());
084            return this;
085        }
086    
087        @Override
088        public boolean isValid() {
089            return value != null;
090        }
091    
092        @Override
093        public DOI verify() throws NumberFormatException {
094            if (value == null) {
095                throw new NumberFormatException();
096            }
097            return this;
098        }
099    
100        @Override
101        public String normalizedValue() {
102            return value != null ? value.toString() : null;
103        }
104    
105        @Override
106        public String format() {
107            return value != null ? value.toString() : null;
108        }
109    
110        public DOI prefix(String prefix) {
111            this.prefix = prefix;
112            return this;
113        }
114    
115        public URI asURI() {
116            return value;
117        }
118    
119        public DOI reset() {
120            this.value = null;
121            return this;
122        }
123    
124        private URI make(String content) {
125            // is it an already a DOI URI?
126            Matcher m = URI_PATTERN.matcher(content);
127            if (m.find()) {
128                String t = content.substring(m.start(), m.end());
129                URI u = URI.create(t);
130                if ("http".equals(u.getScheme()) && ("dx.doi.org".equals(u.getHost()) || "doi.org".equals(u.getHost()))) {
131                    m = PATTERN.matcher(u.getRawPath());
132                    if (m.find()) {
133                        return u;
134                    }
135                }
136            }
137            // not an URI, just the path?
138            m = PATTERN.matcher(content);
139            if (m.find()) {
140                return URI.create(prefix + content.substring(m.start(), m.end()));
141            } else {
142                return null;
143            }
144        }
145    
146    }
