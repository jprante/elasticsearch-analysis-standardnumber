001    package org.xbib.standardnumber;
002    
003    import java.util.regex.Matcher;
004    import java.util.regex.Pattern;
005    
006    /**
007     * ISO 10444 International standard technical report number
008     *
009     * Z39.23 Standard Technical Report Number (STRN)
010     *
011     * Z39.50 BIB-1 Use Attribute 1093
012     *
013     * A registration scheme for a globally unique International Standard Technical Report Number (ISRN)
014     * was standardized in 1994 (ISO 10444), but was never implemented in practice.
015     *
016     * ISO finally withdrew this standard in December 2007.
017     *
018     * Example:
019     *
020     * ISRN UIUCLIS--2001/9+EARCH
021     *
022     * ISRN INRIA/RR--4855--FR+ENG
023     *
024     */
025    public class ISRN implements Comparable<ISRN>, StandardNumber {
026    
027        private static final Pattern PATTERN = Pattern.compile("ISRN [\\p{Graph}\\p{Punct}]{4,36}");
028    
029        private String value;
030    
031        private String formatted;
032    
033        @Override
034        public int compareTo(ISRN isrn) {
035            return isrn != null ? normalizedValue().compareTo(isrn.normalizedValue()) : -1;
036        }
037    
038        @Override
039        public ISRN set(CharSequence value) {
040            this.value = value != null ? value.toString() : null;
041            return this;
042        }
043    
044        @Override
045        public ISRN checksum() {
046            return this;
047        }
048    
049        @Override
050        public ISRN normalize() {
051            Matcher m = PATTERN.matcher(value);
052            if (m.find()) {
053                this.value = clean(value.substring(m.start(), m.end()));
054            }
055            return this;
056        }
057    
058        @Override
059        public ISRN verify() throws NumberFormatException {
060            return this;
061        }
062    
063        @Override
064        public String normalizedValue() {
065            return value;
066        }
067    
068        @Override
069        public String format() {
070            return formatted;
071        }
072    
073        private String clean(String raw) {
074            if (raw == null) {
075                return raw;
076            }
077            StringBuilder sb = new StringBuilder(raw);
078            if (sb.indexOf("ISRN") == 0) {
079                sb = new StringBuilder(sb.substring(4));
080            }
081            this.formatted = "ISRN "
082                    + sb.toString();
083            return sb.toString();
084        }
085    }
