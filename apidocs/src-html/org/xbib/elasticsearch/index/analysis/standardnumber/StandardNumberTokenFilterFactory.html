002package org.xbib.elasticsearch.index.analysis.standardnumber;
003
004import org.apache.lucene.analysis.TokenStream;
005import org.elasticsearch.common.inject.Inject;
006import org.elasticsearch.common.inject.assistedinject.Assisted;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.env.Environment;
009import org.elasticsearch.index.Index;
010import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
011import org.elasticsearch.index.settings.IndexSettings;
012import org.xbib.elasticsearch.index.analysis.standardnumber.Detector;
013
014public class StandardNumberTokenFilterFactory extends AbstractTokenFilterFactory  {
015
016    private final Detector detector;
017
018    @Inject
019    public StandardNumberTokenFilterFactory(Index index,
020                                            @IndexSettings Settings indexSettings, Environment env,
021                                            @Assisted String name, @Assisted Settings settings) {
022        super(index, indexSettings, name, settings);
023        this.detector = new Detector(settings);
024    }
025
026    @Override
027    public TokenStream create(TokenStream tokenStream) {
028        return new StandardNumberTokenFilter(tokenStream, detector);
029    }
030}
