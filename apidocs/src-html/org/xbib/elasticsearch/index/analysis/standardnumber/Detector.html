001package org.xbib.elasticsearch.index.analysis.standardnumber;
002
003import org.elasticsearch.ElasticsearchException;
004import org.elasticsearch.common.collect.Sets;
005import org.elasticsearch.common.component.AbstractLifecycleComponent;
006import org.elasticsearch.common.inject.Inject;
007import org.elasticsearch.common.settings.Settings;
008import org.xbib.standardnumber.ISBN;
009import org.xbib.standardnumber.StandardNumber;
010
011import java.util.Arrays;
012import java.util.Collection;
013import java.util.Set;
014
015import static org.elasticsearch.common.collect.Lists.newLinkedList;
016
017public class Detector extends AbstractLifecycleComponent<Detector>  {
018
019    private final Set<StandardNumber> stdnums;
020
021    @Inject
022    public Detector(Settings settings) {
023        super(settings);
024        String[] s = settings.getAsArray("types", null);
025        Set<String> types = s != null ? Sets.newTreeSet(Arrays.asList(s)) : null;
026        this.stdnums = Sets.newLinkedHashSet();
027        stdnums.addAll(types == null ?
028                StandardNumberService.create() :
029                StandardNumberService.create(types));
030    }
031
032    @Override
033    protected void doStart() throws ElasticsearchException {
034
035    }
036
037    @Override
038    protected void doStop() throws ElasticsearchException {
039
040    }
041
042    @Override
043    protected void doClose() throws ElasticsearchException {
044
045    }
046
047    public Collection<StandardNumber> detect(CharSequence content) {
048        Collection<StandardNumber> candidates = newLinkedList();
049        for (StandardNumber stdnum : stdnums) {
050            stdnum.reset();
051            try {
052                candidates.add(stdnum.set(content).normalize().verify());
053            } catch (NumberFormatException e) {
054                // skip
055            }
056        }
057        return candidates;
058    }
059
060    public Collection<CharSequence> lookup(CharSequence content) {
061        Collection<CharSequence> variants = newLinkedList();
062        for (StandardNumber stdnum : stdnums) {
063            stdnum.reset();
064            if (stdnum instanceof ISBN) {
065                handleISBN((ISBN) stdnum, content, variants);
066            } else {
067                stdnum = stdnum.set(content).normalize();
068                if (stdnum.isValid()) {
069                    variants.add(stdnum.type().toUpperCase() + " " + stdnum.format());
070                    variants.add(stdnum.type().toUpperCase() + " " + stdnum.normalizedValue());
071                }
072            }
073        }
074        return variants;
075    }
076
077    private void handleISBN(ISBN stdnum, CharSequence content, Collection<CharSequence> variants) throws NumberFormatException {
078        ISBN isbn = stdnum.set(content).normalize();
079        if (isbn.isValid()) {
080            if (!isbn.isEAN()) {
081                // create up to 4 variants: ISBN, ISBN normalized, ISBN-13, ISBN-13 normalized
082                variants.add("ISBN " + isbn.ean(false).format());
083                variants.add("ISBN " + isbn.ean(false).normalizedValue());
084                isbn = isbn.ean(true).set(content).normalize();
085                if (isbn.isValid()) {
086                    variants.add("ISBN " + isbn.format());
087                    variants.add("ISBN " + isbn.normalizedValue());
088                }
089            } else {
090                // 2 variants, do not create ISBN-10 for an ISBN-13
091                variants.add("ISBN " + isbn.ean(true).format());
092                variants.add("ISBN " + isbn.ean(true).normalizedValue());
093            }
094        }
095    }
096
097}
