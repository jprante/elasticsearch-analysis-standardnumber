001    package org.xbib.elasticsearch.index.analysis;
002    
003    import org.apache.lucene.analysis.Token;
004    import org.apache.lucene.analysis.TokenFilter;
005    import org.apache.lucene.analysis.TokenStream;
006    import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
007    import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
008    import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
009    import org.apache.lucene.util.AttributeSource;
010    
011    import java.io.IOException;
012    import java.nio.charset.CharacterCodingException;
013    import java.util.Collection;
014    import java.util.LinkedList;
015    
016    public class StandardNumberTokenFilter extends TokenFilter {
017    
018        private final LinkedList<Token> tokens;
019    
020        private final Detector detector;
021    
022        private final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
023    
024        private final OffsetAttribute offsetAtt = addAttribute(OffsetAttribute.class);
025    
026        private final PositionIncrementAttribute posIncAtt = addAttribute(PositionIncrementAttribute.class);
027    
028        private AttributeSource.State current;
029    
030        protected StandardNumberTokenFilter(TokenStream input, Detector detector) {
031            super(input);
032            this.tokens = new LinkedList<Token>();
033            this.detector = detector;
034        }
035    
036        @Override
037        public final boolean incrementToken() throws IOException {
038            if (!tokens.isEmpty()) {
039                assert current != null;
040                Token token = tokens.removeFirst();
041                restoreState(current);
042                termAtt.setEmpty().append(token);
043                offsetAtt.setOffset(token.startOffset(), token.endOffset());
044                posIncAtt.setPositionIncrement(0);
045                return true;
046            }
047            if (input.incrementToken()) {
048                detectAll();
049                if (!tokens.isEmpty()) {
050                    current = captureState();
051                }
052                return true;
053            } else {
054                return false;
055            }
056        }
057    
058        protected void detectAll() throws CharacterCodingException {
059            CharSequence term = new String(termAtt.buffer(), 0, termAtt.length());
060            Collection<CharSequence> variants = detector.lookup(term);
061            int start = offsetAtt.startOffset();
062            for (CharSequence ch : variants) {
063                if (ch != null) {
064                    String s = ch.toString();
065                    tokens.add(new Token(s, start, start + s.length()));
066                }
067            }
068        }
069    
070        @Override
071        public void reset() throws IOException {
072            super.reset();
073            tokens.clear();
074            current = null;
075        }
076    }
