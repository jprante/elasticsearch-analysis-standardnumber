001    package org.xbib.elasticsearch.index.analysis;
002    
003    import org.apache.lucene.analysis.TokenStream;
004    import org.elasticsearch.common.inject.Inject;
005    import org.elasticsearch.common.inject.assistedinject.Assisted;
006    import org.elasticsearch.common.settings.Settings;
007    import org.elasticsearch.env.Environment;
008    import org.elasticsearch.index.Index;
009    import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
010    import org.elasticsearch.index.settings.IndexSettings;
011    
012    public class ConcatTokenFilterFactory extends AbstractTokenFilterFactory  {
013    
014        @Inject
015        public ConcatTokenFilterFactory(Index index,
016                                            @IndexSettings Settings indexSettings, Environment env,
017                                            @Assisted String name, @Assisted Settings settings) {
018            super(index, indexSettings, name, settings);
019        }
020    
021        @Override
022        public TokenStream create(TokenStream tokenStream) {
023            return new ConcatTokenFilter(tokenStream);
024    
025        }
026    }
