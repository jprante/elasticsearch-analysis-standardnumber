001    package org.xbib.elasticsearch.index.analysis;
002    
003    import java.io.IOException;
004    import java.util.LinkedList;
005    import java.util.List;
006    
007    import org.apache.lucene.analysis.TokenFilter;
008    import org.apache.lucene.analysis.TokenStream;
009    import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
010    import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
011    import org.apache.lucene.util.AttributeSource;
012    
013    public class ConcatTokenFilter extends TokenFilter {
014    
015        private CharTermAttribute termAttr;
016        private PositionIncrementAttribute posIncAttr;
017    
018        private AttributeSource.State current;
019        private LinkedList<List<String>> words;
020        private LinkedList<String> phrases;
021    
022        private boolean concat = false;
023    
024        protected ConcatTokenFilter(TokenStream input) {
025            super(input);
026            this.termAttr = addAttribute(CharTermAttribute.class);
027            this.posIncAttr = addAttribute(PositionIncrementAttribute.class);
028            this.words = new LinkedList<List<String>>();
029            this.phrases = new LinkedList<String>();
030        }
031    
032        @Override
033        public boolean incrementToken() throws IOException {
034            while (input.incrementToken()) {
035                String term = new String(termAttr.buffer(), 0, termAttr.length());
036                List<String> word = posIncAttr.getPositionIncrement() > 0 ?
037                        new LinkedList<String>() : words.removeLast();
038                word.add(term);
039                words.add(word);
040            }
041            // now write out as a single token
042            if (!concat) {
043                makePhrases(words, phrases, 0);
044                concat = true;
045            }
046            while (phrases.size() > 0) {
047                String phrase = phrases.removeFirst();
048                restoreState(current);
049                clearAttributes();
050                termAttr.copyBuffer(phrase.toCharArray(), 0, phrase.length());
051                termAttr.setLength(phrase.length());
052                current = captureState();
053                return true;
054            }
055            concat = false;
056            return false;
057        }
058    
059        private void makePhrases(List<List<String>> words, List<String> phrases, int currPos) {
060            for (int i = currPos; i < words.size(); i++) {
061                if (phrases.size() == 0) {
062                    phrases.addAll(words.get(i));
063                } else {
064                    List<String> newPhrases = new LinkedList<String>();
065                    for (String phrase : phrases) {
066                        for (String word : words.get(i)) {
067                            newPhrases.add(phrase + " " + word);
068                        }
069                    }
070                    phrases.clear();
071                    phrases.addAll(newPhrases);
072                }
073            }
074        }
075    }
