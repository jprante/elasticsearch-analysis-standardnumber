001    package org.xbib.elasticsearch.index.analysis;
002    
003    import org.xbib.standardnumber.ISBN;
004    import org.xbib.standardnumber.StandardNumber;
005    
006    import java.util.Collection;
007    
008    import static org.elasticsearch.common.collect.Lists.newLinkedList;
009    
010    public class Detector {
011    
012        public Collection<StandardNumber> detect(CharSequence content) {
013            Collection<StandardNumber> candidates = newLinkedList();
014            try {
015                ISBN isbn = new ISBN().set(content).normalize().verify();
016                candidates.add(isbn);
017            } catch (NumberFormatException e) {
018                // ignore ISBN
019            }
020            return candidates;
021        }
022    
023        public Collection<CharSequence> lookup(CharSequence content) {
024            Collection<CharSequence> variants = newLinkedList();
025            try {
026                ISBN isbn = new ISBN().set(content).normalize().verify();
027                if (!isbn.isEAN()) {
028                    // create up to 4 variants: ISBN, ISBN normalized, ISBN-13, ISBN-13 normalized
029                    variants.add(isbn.ean(false).format());
030                    variants.add(isbn.ean(false).normalizedValue());
031                    isbn = isbn.ean(true).set(content).normalize().verify();
032                    variants.add(isbn.format());
033                    variants.add(isbn.normalizedValue());
034                } else {
035                    // 2 variants, do not create ISBN-10 for an ISBN-13
036                    variants.add(isbn.ean(true).format());
037                    variants.add(isbn.ean(true).normalizedValue());
038                }
039            } catch (NumberFormatException e) {
040                // ignore ISBN
041            }
042            return variants;
043        }
044    }
