001    package org.xbib.elasticsearch.index.analysis;
002    
003    import org.elasticsearch.common.collect.Sets;
004    import org.elasticsearch.common.settings.Settings;
005    import org.xbib.standardnumber.ISBN;
006    import org.xbib.standardnumber.StandardNumber;
007    
008    import java.util.Arrays;
009    import java.util.Collection;
010    import java.util.Set;
011    
012    import static org.elasticsearch.common.collect.Lists.newLinkedList;
013    
014    public class Detector {
015    
016        private final Set<StandardNumber> stdnums;
017    
018        public Detector(Settings settings) {
019            String[] s = settings.getAsArray("types", null);
020            Set<String> types = s != null ? Sets.newTreeSet(Arrays.asList(s)) : null;
021            this.stdnums = Sets.newLinkedHashSet();
022            stdnums.addAll(types == null ?
023                    StandardNumberService.create() :
024                    StandardNumberService.create(types));
025        }
026    
027        public Collection<StandardNumber> detect(CharSequence content) {
028            Collection<StandardNumber> candidates = newLinkedList();
029            for (StandardNumber stdnum : stdnums) {
030                stdnum.reset();
031                try {
032                    candidates.add(stdnum.set(content).normalize().verify());
033                } catch (NumberFormatException e) {
034                    // skip
035                }
036            }
037            return candidates;
038        }
039    
040        public Collection<CharSequence> lookup(CharSequence content) {
041            Collection<CharSequence> variants = newLinkedList();
042            for (StandardNumber stdnum : stdnums) {
043                stdnum.reset();
044                if (stdnum instanceof ISBN) {
045                    lookupISBN(content, variants);
046                } else {
047                    stdnum = stdnum.set(content).normalize();
048                    if (stdnum.isValid()) {
049                        variants.add(stdnum.type().toUpperCase() + " " + stdnum.format());
050                        variants.add(stdnum.type().toUpperCase() + " " + stdnum.normalizedValue());
051                    }
052                }
053            }
054            return variants;
055        }
056    
057        private void lookupISBN(CharSequence content, Collection<CharSequence> variants) throws NumberFormatException {
058            ISBN isbn = new ISBN().set(content).normalize();
059            if (!isbn.isValid()) {
060                return;
061            }
062            if (!isbn.isEAN()) {
063                // create up to 4 variants: ISBN, ISBN normalized, ISBN-13, ISBN-13 normalized
064                if (isbn.isValid()) {
065                    variants.add("ISBN " + isbn.ean(false).format());
066                    variants.add("ISBN " + isbn.ean(false).normalizedValue());
067                }
068                isbn = isbn.ean(true).set(content).normalize();
069                if (isbn.isValid()) {
070                    variants.add("ISBN " + isbn.format());
071                    variants.add("ISBN " + isbn.normalizedValue());
072                }
073            } else {
074                // 2 variants, do not create ISBN-10 for an ISBN-13
075                if (isbn.isValid()) {
076                    variants.add("ISBN " + isbn.ean(true).format());
077                    variants.add("ISBN " + isbn.ean(true).normalizedValue());
078                }
079            }
080        }
081    
082    }
