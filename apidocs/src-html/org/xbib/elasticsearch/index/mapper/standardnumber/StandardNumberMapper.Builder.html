002    package org.xbib.elasticsearch.index.mapper.standardnumber;
003    
004    import org.elasticsearch.common.xcontent.XContentBuilder;
005    import org.elasticsearch.common.xcontent.XContentParser;
006    import org.elasticsearch.index.mapper.FieldMapperListener;
007    import org.elasticsearch.index.mapper.Mapper;
008    import org.elasticsearch.index.mapper.MapperParsingException;
009    import org.elasticsearch.index.mapper.MergeContext;
010    import org.elasticsearch.index.mapper.MergeMappingException;
011    import org.elasticsearch.index.mapper.ObjectMapperListener;
012    import org.elasticsearch.index.mapper.ParseContext;
013    import org.elasticsearch.index.mapper.core.StringFieldMapper;
014    import org.xbib.elasticsearch.index.analysis.Detector;
015    import org.xbib.standardnumber.StandardNumber;
016    
017    import java.io.IOException;
018    import java.nio.charset.Charset;
019    import java.util.Collection;
020    import java.util.List;
021    import java.util.Map;
022    
023    import static org.elasticsearch.index.mapper.MapperBuilders.stringField;
024    
025    public class StandardNumberMapper implements Mapper {
026    
027        public static final String CONTENT_TYPE = "standardnumber";
028    
029        public static class Builder extends Mapper.Builder<Builder, StandardNumberMapper> {
030    
031            private StringFieldMapper.Builder contentBuilder;
032    
033            private StringFieldMapper.Builder stdnumBuilder = stringField("stdnum");
034    
035            private Detector detector;
036    
037            public Builder(String name, Detector detector) {
038                super(name);
039                this.detector = detector;
040                this.contentBuilder = stringField(name);
041                this.builder = this;
042            }
043    
044            public Builder content(StringFieldMapper.Builder content) {
045                this.contentBuilder = content;
046                return this;
047            }
048    
049            public Builder lang(StringFieldMapper.Builder lang) {
050                this.stdnumBuilder = lang;
051                return this;
052            }
053    
054            @Override
055            public StandardNumberMapper build(BuilderContext context) {
056                context.path().add(name);
057                StringFieldMapper contentMapper = contentBuilder.build(context);
058                StringFieldMapper langMapper = stdnumBuilder.build(context);
059                context.path().remove();
060                return new StandardNumberMapper(name, detector, contentMapper, langMapper);
061            }
062        }
063    
064        public static class TypeParser implements Mapper.TypeParser {
065    
066            private Detector detector;
067    
068            public TypeParser(Detector detector) {
069                this.detector = detector;
070            }
071    
072            @SuppressWarnings({"unchecked","rawtypes"})
073            @Override
074            public Mapper.Builder parse(String name, Map<String, Object> node, ParserContext parserContext)
075                    throws MapperParsingException {
076                StandardNumberMapper.Builder builder = new Builder(name, detector);
077                for (Map.Entry<String, Object> entry : node.entrySet()) {
078                    String fieldName = entry.getKey();
079                    Object fieldNode = entry.getValue();
080    
081                    if (fieldName.equals("fields")) {
082                        Map<String, Object> fieldsNode = (Map<String, Object>) fieldNode;
083                        for (Map.Entry<String, Object> fieldsEntry : fieldsNode.entrySet()) {
084                            String propName = fieldsEntry.getKey();
085                            Object propNode = fieldsEntry.getValue();
086    
087                            if (name.equals(propName)) {
088                                builder.content((StringFieldMapper.Builder) parserContext.typeParser("string").parse(name,
089                                        (Map<String, Object>) propNode, parserContext));
090                            } else if ("stdnum".equals(propName)) {
091                                builder.lang((StringFieldMapper.Builder) parserContext.typeParser("string").parse("lang",
092                                        (Map<String, Object>) propNode, parserContext));
093                            }
094                        }
095                    }
096                }
097    
098                return builder;
099            }
100        }
101    
102        private final String name;
103        private final Detector detector;
104        private final StringFieldMapper contentMapper;
105        private final StringFieldMapper stdnumMapper;
106    
107        public StandardNumberMapper(String name, Detector detector,
108                                    StringFieldMapper contentMapper,
109                                    StringFieldMapper stdnumMapper) {
110            this.name = name;
111            this.detector = detector;
112            this.contentMapper = contentMapper;
113            this.stdnumMapper = stdnumMapper;
114        }
115    
116        @Override
117        public String name() {
118            return name;
119        }
120    
121        @Override
122        public void parse(ParseContext context) throws IOException {
123            String content = null;
124    
125            XContentParser parser = context.parser();
126            XContentParser.Token token = parser.currentToken();
127    
128            if (token == XContentParser.Token.VALUE_STRING) {
129                // try decode UTF-8 base64 (e.g. from attachment mapper plugin)
130                content = parser.text();
131                try {
132                    byte[] b = parser.binaryValue();
133                    if (b != null && b.length > 0) {
134                        content = new String(b, Charset.forName("UTF-8"));
135                    }
136                } catch (Exception e) {
137                }
138            }
139    
140            context.externalValue(content);
141            contentMapper.parse(context);
142    
143            try {
144                Collection<StandardNumber> stdnums = detector.detect(content);
145                for (StandardNumber stdnum : stdnums) {
146                    context.externalValue(stdnum.normalizedValue());
147                    stdnumMapper.parse(context);
148                }
149            } catch(NumberFormatException e) {
150                context.externalValue("unknown");
151                stdnumMapper.parse(context);
152            }
153        }
154    
155        @Override
156        public void merge(Mapper mergeWith, MergeContext mergeContext) throws MergeMappingException {
157        }
158    
159        @Override
160        public void traverse(FieldMapperListener fieldMapperListener) {
161            contentMapper.traverse(fieldMapperListener);
162            stdnumMapper.traverse(fieldMapperListener);
163        }
164    
165        @Override
166        public void traverse(ObjectMapperListener objectMapperListener) {
167        }
168    
169        @Override
170        public void close() {
171            contentMapper.close();
172            stdnumMapper.close();
173        }
174    
175        @Override
176        public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
177            builder.startObject(name);
178            builder.field("type", CONTENT_TYPE);
179    
180            builder.startObject("fields");
181            contentMapper.toXContent(builder, params);
182            stdnumMapper.toXContent(builder, params);
183            builder.endObject();
184    
185            builder.endObject();
186            return builder;
187        }
188    }
