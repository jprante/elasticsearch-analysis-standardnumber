001    package org.xbib.elasticsearch.plugin.analysis.standardnumber;
002    
003    import org.elasticsearch.index.analysis.AnalysisModule;
004    import org.elasticsearch.plugins.AbstractPlugin;
005    import org.xbib.elasticsearch.index.analysis.ConcatTokenFilterFactory;
006    import org.xbib.elasticsearch.index.analysis.StandardNumberTokenFilterFactory;
007    
008    public class AnalysisStandardNumberPlugin extends AbstractPlugin {
009    
010        @Override
011        public String name() {
012            return "analysis-standardnumber";
013        }
014    
015        @Override
016        public String description() {
017            return "Analysis plugin for standard numbers";
018        }
019    
020        public void onModule(AnalysisModule module) {
021            module.addTokenFilter("standardnumber", StandardNumberTokenFilterFactory.class);
022            module.addTokenFilter("concat", ConcatTokenFilterFactory.class);
023        }
024    
025    }
