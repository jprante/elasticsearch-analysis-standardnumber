001package org.xbib.elasticsearch.plugin.analysis.standardnumber;
002
003import org.elasticsearch.common.inject.Inject;
004import org.elasticsearch.common.settings.Settings;
005import org.elasticsearch.index.AbstractIndexComponent;
006import org.elasticsearch.index.Index;
007import org.elasticsearch.index.mapper.MapperService;
008import org.elasticsearch.index.settings.IndexSettings;
009import org.xbib.elasticsearch.index.analysis.standardnumber.Detector;
010import org.xbib.elasticsearch.index.mapper.standardnumber.StandardNumberMapper;
011
012public class RegisterStandardNumberType extends AbstractIndexComponent {
013
014    @Inject
015    public RegisterStandardNumberType(Index index, @IndexSettings Settings indexSettings,
016                                      MapperService mapperService, Detector detector) {
017        super(index, indexSettings);
018        mapperService.documentMapperParser().putTypeParser("standardnumber",
019                new StandardNumberMapper.TypeParser(detector));
020    }
021}
