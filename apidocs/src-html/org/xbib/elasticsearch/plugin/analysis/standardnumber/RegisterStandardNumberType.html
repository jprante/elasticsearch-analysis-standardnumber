001    package org.xbib.elasticsearch.plugin.analysis.standardnumber;
002    
003    import org.elasticsearch.common.inject.Inject;
004    import org.elasticsearch.common.settings.Settings;
005    import org.elasticsearch.index.AbstractIndexComponent;
006    import org.elasticsearch.index.Index;
007    import org.elasticsearch.index.mapper.MapperService;
008    import org.elasticsearch.index.settings.IndexSettings;
009    import org.xbib.elasticsearch.index.analysis.Detector;
010    import org.xbib.elasticsearch.index.mapper.standardnumber.StandardNumberMapper;
011    
012    public class RegisterStandardNumberType extends AbstractIndexComponent {
013    
014        @Inject
015        public RegisterStandardNumberType(Index index, @IndexSettings Settings indexSettings,
016                                          MapperService mapperService, Detector detector) {
017            super(index, indexSettings);
018            mapperService.documentMapperParser().putTypeParser("stdnum",
019                    new StandardNumberMapper.TypeParser(detector));
020        }
021    }
